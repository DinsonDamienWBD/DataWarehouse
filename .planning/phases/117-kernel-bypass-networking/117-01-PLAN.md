```csharp
[SdkCompatibility("8.0.0", Notes = "Phase 117: Kernel bypass networking (SIO-06)")]
public interface IKernelBypassTransport : IRdmaTransport, IAsyncDisposable
{
    KernelBypassCapability Capabilities { get; }
    TransportPreference TransportType { get; }

    // Message-oriented send/receive (unlike IRdmaTransport's address-based read/write)
    Task<int> SendAsync(Memory<byte> data, CancellationToken ct = default);
    Task<int> ReceiveAsync(Memory<byte> buffer, CancellationToken ct = default);

    // Explicit memory registration for zero-copy DMA
    MemoryRegionHandle RegisterMemory(Memory<byte> region);
    void DeregisterMemory(MemoryRegionHandle handle);

    // Bulk transfer with pre-registered memory
    Task<int> SendRegisteredAsync(MemoryRegionHandle handle, int offset, int length, CancellationToken ct = default);
    Task<int> ReceiveRegisteredAsync(MemoryRegionHandle handle, int offset, int length, CancellationToken ct = default);
}
```

Create `KernelBypassTypes.cs` in same directory with:

- `KernelBypassCapability` [Flags] enum: `None=0, ZeroCopy=1, PreRegisteredMemory=2, PollMode=4, Interrupt=8, Batching=16`
- `MemoryRegionHandle` sealed class: `nint Address`, `int Length`, `ulong RegistrationKey`, implements `IDisposable` with Action-based cleanup callback
