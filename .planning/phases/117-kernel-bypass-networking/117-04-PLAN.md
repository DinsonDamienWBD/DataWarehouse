```csharp
[SdkCompatibility("8.0.0", Notes = "Phase 117: Fabric link manager (SIO-06)")]
public sealed class FabricLinkManager : IAsyncDisposable
{
    private readonly ConcurrentDictionary<string, FabricLink> _links = new();
    private readonly FabricLinkManagerConfig _config;
    private readonly Func<TransportPreference, IKernelBypassTransport> _transportFactory;
    private readonly IFederationOrchestrator? _federationOrchestrator;
    private readonly ILogger? _logger;
    private readonly Timer _heartbeatTimer;
    private readonly string _localNodeId;

    public FabricLinkManager(
        string localNodeId,
        FabricLinkManagerConfig config,
        Func<TransportPreference, IKernelBypassTransport> transportFactory,
        IFederationOrchestrator? federationOrchestrator = null,
        ILogger? logger = null);

    // Core operations
    public async Task<FabricLink> ConnectPeerAsync(PeerEndpoint peer, CancellationToken ct = default);
    public async Task DisconnectPeerAsync(string nodeId, CancellationToken ct = default);
    public FabricLinkSnapshot? GetLink(string nodeId);
    public IReadOnlyList<FabricLinkSnapshot> GetAllLinks();

    // Send/receive through managed link
    public async Task<int> SendToPeerAsync(string nodeId, Memory<byte> data, CancellationToken ct = default);
    public async Task<int> ReceiveFromPeerAsync(string nodeId, Memory<byte> buffer, CancellationToken ct = default);

    // Transport management
    private TransportPreference SelectTransport(PeerEndpoint peer);
    private async Task HeartbeatLoopAsync();
    private async Task ReconnectAsync(FabricLink link, CancellationToken ct);
    private async Task TryUpgradeTransportAsync(FabricLink link);
}
```

Key implementation:
- `SelectTransport`: Check `_config.ForcedTransport` first. Otherwise call `KernelBypassPlatformDetector.DetectAvailable()` and pick best that peer also supports. Respect `PeerEndpoint.PreferredTransport`.
- `ConnectPeerAsync`: Create transport via factory, call `ConnectAsync`, create FabricLink, register with federation orchestrator if available, start tracking.
- Heartbeat: `Timer` fires every `HeartbeatInterval`. For each link, send 8-byte heartbeat ping. If no response within `HeartbeatTimeout`, mark `Degraded`. If 3 consecutive misses, mark `Failed` and trigger reconnect.
- Reconnect: Exponential backoff with jitter. Try same transport first, then downgrade (RDMA->DPDK->TCP) if `AllowTransportDowngrade`. Cap at `MaxReconnectAttempts`.
- Upgrade: If link is on TCP/DPDK and `AllowTransportUpgrade`, periodically (every 60s) check if better transport became available. If so, create new transport, migrate link, dispose old.
- `SendToPeerAsync/ReceiveFromPeerAsync`: Look up link, validate Connected state, delegate to `link.Transport.SendAsync/ReceiveAsync`, update byte counters.
- Thread-safety: All link mutations under per-link SemaphoreSlim. ConcurrentDictionary for link lookup.

Tests inject mock `IKernelBypassTransport` via the factory delegate. Use `TaskCompletionSource` to control heartbeat timing. Verify all 10 behaviors including transport cascade, backoff timing, and upgrade path.
