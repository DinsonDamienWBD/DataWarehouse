```csharp
public enum MessagePlane { Control, Data }

public sealed record NetworkMessage
{
    public required string MessageId { get; init; }
    public required string Topic { get; init; }
    public required MessagePlane Plane { get; init; }
    public required byte[] Payload { get; init; }
    public required string SourceNodeId { get; init; }
    public string? TargetNodeId { get; init; }  // null = broadcast
    public DateTimeOffset Timestamp { get; init; } = DateTimeOffset.UtcNow;
    public Dictionary<string, string> Headers { get; init; } = new();
}

public static class MessageFraming
{
    // Wire format: [4-byte length][payload]
    // Length is big-endian uint32, max 64MB per message
    public const int HeaderSize = 4;
    public const int MaxPayloadSize = 64 * 1024 * 1024;

    public static int WriteFrame(Span<byte> buffer, ReadOnlySpan<byte> payload);
    public static bool TryReadFrame(ReadOnlySpan<byte> buffer, out ReadOnlySpan<byte> payload, out int bytesConsumed);
    public static int FrameSize(int payloadLength) => HeaderSize + payloadLength;
}

public sealed record DataPlaneConfig
{
    // Topics routed over data plane (RDMA/DPDK) -- everything else goes control plane (TCP)
    public HashSet<string> DataPlaneTopics { get; init; } = new()
    {
        "vde.block.replicate",
        "vde.shard.transfer",
        "vde.extent.sync",
        "storage.bulk.transfer"
    };

    public HashSet<string> ControlPlaneTopics { get; init; } = new()
    {
        "federation.heartbeat",
        "federation.topology",
        "cluster.membership",
        "metadata.sync"
    };

    // If topic not in either set, route based on payload size
    public int DataPlanePayloadThreshold { get; init; } = 4096;  // >4KB goes data plane
}
```

```csharp
[SdkCompatibility("8.0.0", Notes = "Phase 117: Cross-VDE message bus bridge (SIO-06)")]
public sealed class NetworkMessageBusBridge : FederatedMessageBusBase
{
    private readonly FabricLinkManager _linkManager;
    private readonly DataPlaneConfig _dataPlaneConfig;
    private readonly ILogger? _logger;
    private readonly CancellationTokenSource _receiveCts = new();
    private Task? _receiveLoop;

    public NetworkMessageBusBridge(
        IMessageBus localBus,
        IClusterMembership clusterMembership,
        FabricLinkManager linkManager,
        DataPlaneConfig? dataPlaneConfig = null,
        ILogger? logger = null)
        : base(localBus, clusterMembership);

    // Override PublishAsync to route by plane
    public override async Task PublishAsync(string topic, PluginMessage message, CancellationToken ct = default)
    {
        var plane = ClassifyTopic(topic, message.Data?.Length ?? 0);
        if (plane == MessagePlane.Data)
            await SendViaDataPlaneAsync(topic, message, ct);
        else
            await base.PublishAsync(topic, message, ct);  // Control plane via base (TCP)
    }

    private MessagePlane ClassifyTopic(string topic, int payloadSize);
    private async Task SendViaDataPlaneAsync(string topic, PluginMessage message, CancellationToken ct);
    private async Task ReceiveLoopAsync(CancellationToken ct);  // Background: poll FabricLinkManager for incoming
    public void StartReceiving();
    public void StopReceiving();
}
```

Key implementation:
- `ClassifyTopic`: Check `DataPlaneConfig.DataPlaneTopics` first, then `ControlPlaneTopics`, then fall back to payload size threshold.
- `SendViaDataPlaneAsync`: Serialize PluginMessage to NetworkMessage, frame with MessageFraming, send via `FabricLinkManager.SendToPeerAsync`. If TargetNodeId is null, iterate all connected peers (broadcast).
- `ReceiveLoopAsync`: Background task that polls each connected peer via `FabricLinkManager.ReceiveFromPeerAsync`, deserializes NetworkMessage, dispatches to local subscribers via `LocalBus.PublishAsync`.
- Serialization: Use `System.Text.Json` for NetworkMessage envelope, raw bytes for payload. Topic + headers in envelope, payload passed through.
