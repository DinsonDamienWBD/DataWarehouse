 521            }
 522        }

 562    private (bool IsMatch, List<InferenceFact> MatchedFacts, double Confidence) EvaluateRule(
 563        InferenceRule rule,
 564        Dictionary<string, List<InferenceFact>> factLookup,
 565        List<InferenceFact> allFacts)
 566    {
 567        var matchedFacts = new List<InferenceFact>();
 568        var matchedCount = 0;
 569        var totalWeight = 0.0;
 570        var weightedConfidence = 0.0;
 571
 572        foreach (var condition in rule.Conditions)
 573        {
 574            var conditionMet = EvaluateCondition(condition, factLookup, allFacts, out var matchedFact);
 575            if (conditionMet && matchedFact != null)
 576            {
 577                matchedFacts.Add(matchedFact);
 578                matchedCount++;
 579                totalWeight += condition.Weight;
 580                weightedConfidence += condition.Weight * matchedFact.Confidence;
 581            }
 582        }
 583
 584        var requiredMatches = rule.MinConditionsToMatch == -1
 585            ? rule.Conditions.Count
 586            : rule.MinConditionsToMatch;
 587
 588        var isMatch = matchedCount >= requiredMatches;
 589        var confidence = isMatch && totalWeight > 0
 590            ? (weightedConfidence / totalWeight) * rule.BaseConfidence
 591            : 0.0;
 592
 593        return (isMatch, matchedFacts, confidence);
 594    }

 596    private bool EvaluateCondition(
 597        RuleCondition condition,
 598        Dictionary<string, List<InferenceFact>> factLookup,
 599        List<InferenceFact> allFacts,
 600        out InferenceFact? matchedFact)
 601    {
 602        matchedFact = null;
 603
 604        // Handle existence checks
 605        if (condition.Operator == ConditionOperator.Exists)
 606        {
 607            if (factLookup.TryGetValue(condition.FactType, out var facts) && facts.Count > 0)
 608            {
 609                matchedFact = facts[0];
 610                return true;
 611            }
 612            return false;
 613        }
 614
 615        if (condition.Operator == ConditionOperator.NotExists)
 616        {
 617            return !factLookup.ContainsKey(condition.FactType);
 618        }
 619
 620        // Get the fact to evaluate
 621        if (!factLookup.TryGetValue(condition.FactType, out var matchFacts) || matchFacts.Count == 0)
 622        {
 623            return false;
 624        }
 625
 626        var fact = matchFacts[0];
 627        var compareValue = condition.Value;
 628
 629        // Handle relative comparisons
 630        if (!string.IsNullOrEmpty(condition.RelativeToFact))
 631        {
 632            if (!factLookup.TryGetValue(condition.RelativeToFact, out var relativeFacts) || relativeFacts.Count == 0)
 633            {
 634                return false;
 635            }
 636            compareValue = relativeFacts[0].Value;
 637        }
 638
 639        var result = EvaluateOperator(condition.Operator, fact.Value, compareValue);
 640        if (result)
 641        {
 642            matchedFact = fact;
 643        }
 644
 645        return result;
 646    }

 648    private static bool EvaluateOperator(ConditionOperator op, object? factValue, object? compareValue)
 649    {
 650        return op switch
 651        {
 652            ConditionOperator.Equals => Equals(factValue, compareValue),
 653            ConditionOperator.NotEquals => !Equals(factValue, compareValue),
 654            ConditionOperator.GreaterThan => Compare(factValue, compareValue) > 0,
 655            ConditionOperator.LessThan => Compare(factValue, compareValue) < 0,
 656            ConditionOperator.GreaterThanOrEqual => Compare(factValue, compareValue) >= 0,
 657            ConditionOperator.LessThanOrEqual => Compare(factValue, compareValue) <= 0,
 658            ConditionOperator.Contains => ContainsValue(factValue, compareValue),
 659            ConditionOperator.NotContains => !ContainsValue(factValue, compareValue),
 660            ConditionOperator.Matches => MatchesPattern(factValue, compareValue),
 661            ConditionOperator.After => CompareDateTime(factValue, compareValue) > 0,
 662            ConditionOperator.Before => CompareDateTime(factValue, compareValue) < 0,
 663            ConditionOperator.Deviates => CheckDeviation(factValue, compareValue),
 664            _ => false
 665        };
 666    }
