 510        EmbeddingRequest request,
 511        string? providerName = null,
 512        CancellationToken ct = default)
 513    {
 514        ObjectDisposedException.ThrowIf(_disposed, this);
 515        ArgumentNullException.ThrowIfNull(request);
 516
 517        providerName ??= _config.DefaultEmbeddingProvider;
 518        var provider = _providerResolver(providerName);
 519        if (provider == null)
 520        {
 521            throw new InvalidOperationException($"Provider '{providerName}' not found.");
 522        }
 523
 524        // Get inputs
 525        var inputs = request.Inputs ?? (request.Input != null ? new[] { request.Input } : Array.Empty<string>());
 526        if (inputs.Length == 0)
 527        {
 528            throw new ArgumentException("Either Input or Inputs must be provided.");
 529        }
 530
 531        // Generate embeddings using batch API
 532        var embeddings = await provider.GetEmbeddingsBatchAsync(inputs, ct);
 533
 534        // Convert to response format
 535        var data = new List<EmbeddingData>();
 536        for (int i = 0; i < embeddings.Length; i++)
 537        {
 538            data.Add(new EmbeddingData
 539            {
 540                Index = i,
 541                Embedding = embeddings[i]
 542            });
 543        }
 544
 545        // Estimate token count (rough: ~1 token per 4 characters)
 546        var totalTokens = inputs.Sum(s => s.Length / 4);
 547
 548        return new EmbeddingResponse
 549        {
 550            Data = data,
 551            Model = request.Model,
 552            TotalTokens = totalTokens
 553        };
 554    }

 556    /// <summary>
 557    /// Handles a streaming chat request.
 558    /// </summary>
 559    /// <param name="request">The chat request.</param>
 560    /// <param name="providerName">Optional provider name.</param>
 561    /// <param name="ct">Cancellation token.</param>
 562    /// <returns>Async enumerable of response chunks.</returns>
 563    public async IAsyncEnumerable<string> HandleStreamAsync(
 564        ChatRequest request,
 565        string? providerName = null,
 566        [EnumeratorCancellation] CancellationToken ct = default)
 567    {
 568        ObjectDisposedException.ThrowIf(_disposed, this);
 569        ArgumentNullException.ThrowIfNull(request);
 570
 571        providerName ??= _config.DefaultProvider;
 572        var provider = _providerResolver(providerName);
 573        if (provider == null)
 574        {
 575            throw new InvalidOperationException($"Provider '{providerName}' not found.");
 576        }
 577
 578        // Build messages
 579        List<ChatMessage> messages;
 580        if (!string.IsNullOrEmpty(request.ConversationId))
 581        {
 582            var conversation = _conversationManager.GetOrCreate(request.ConversationId);
 583            messages = conversation.BuildMessages(request.Messages, _config.MaxHistoryTokens);
 584        }
 585        else
 586        {
 587            messages = request.Messages;
 588        }
 589
 590        var prompt = BuildPromptFromMessages(messages);
 591        var systemMessage = ExtractSystemMessage(messages);
 592
 593        // Stream response
 594        var fullContent = new StringBuilder();
 595        await foreach (var chunk in _streamingHandler.StreamAsync(provider, prompt, systemMessage, request.Model, ct))
 596        {
 597            fullContent.Append(chunk);
 598            yield return chunk;
 599        }
 600
 601        // Update conversation after streaming completes
 602        if (!string.IsNullOrEmpty(request.ConversationId))
 603        {
 604            var conversation = _conversationManager.GetOrCreate(request.ConversationId);
 605            conversation.AddUserMessage(messages.Last().Content ?? string.Empty);
 606            conversation.AddAssistantMessage(fullContent.ToString());
 607        }
 608    }
 609
 610    private string BuildPromptFromMessages(List<ChatMessage> messages)
 611    {
 612        var sb = new StringBuilder();
 613        foreach (var msg in messages.Where(m => m.Role != "system"))
 614        {
 615            var prefix = msg.Role == "user" ? "User" : "Assistant";
 616            sb.AppendLine($"{prefix}: {msg.Content}");
 617        }
 618        return sb.ToString().TrimEnd();
 619    }

 621    private string? ExtractSystemMessage(List<ChatMessage> messages)
 622    {
 623        return messages.FirstOrDefault(m => m.Role == "system")?.Content;
 624    }

 628    {
 629        if (_disposed) return;
 630        _disposed = true;
 631        _conversationManager.Dispose();
 632    }

 639/// <summary>
 640/// Manages conversation state with automatic cleanup and history trimming.
 641/// Supports up to 10,000 concurrent conversations with 24-hour TTL.
 642/// </summary>
 643public sealed class ConversationManager : IDisposable
 644{
 645    private readonly ConcurrentDictionary<string, Conversation> _conversations = new();
 646    private readonly Timer _cleanupTimer;
 647    private readonly TimeSpan _ttl;
 648    private readonly int _maxConversations;
 649    private bool _disposed;
 650
 651    /// <summary>
 652    /// Initializes a new instance of the <see cref="ConversationManager"/> class.
 653    /// </summary>
 654    /// <param name="ttl">Time-to-live for inactive conversations.</param>
 655    /// <param name="maxConversations">Maximum number of conversations to track.</param>
 656    public ConversationManager(TimeSpan ttl, int maxConversations)
 657    {
 658        _ttl = ttl;
 659        _maxConversations = maxConversations;
 660
 661        // Periodic cleanup every 5 minutes
 662        _cleanupTimer = new Timer(
 663            CleanupExpired,
 664            null,
 665            TimeSpan.FromMinutes(5),
 666            TimeSpan.FromMinutes(5));
 667    }
 668
 669    /// <summary>
 670    /// Gets or creates a conversation by ID.
 671    /// </summary>
 672    /// <param name="conversationId">The conversation identifier.</param>
 673    /// <returns>The conversation instance.</returns>
 674    public Conversation GetOrCreate(string conversationId)
 675    {
 676        ObjectDisposedException.ThrowIf(_disposed, this);
 677        ArgumentException.ThrowIfNullOrWhiteSpace(conversationId);
 678
 679        var conversation = _conversations.GetOrAdd(conversationId, id => new Conversation(id));
 680        conversation.Touch();
 681
 682        // Enforce max conversations
 683        while (_conversations.Count > _maxConversations)
 684        {
 685            var oldest = _conversations
 686                .OrderBy(kvp => kvp.Value.LastAccessedAt)
 687                .FirstOrDefault();
 688
 689            if (!string.IsNullOrEmpty(oldest.Key))
 690            {
 691                _conversations.TryRemove(oldest.Key, out _);
 692            }
 693            else
 694            {
 695                break;
 696            }
 697        }
 698
 699        return conversation;
 700    }
 701
 702    /// <summary>
 703    /// Removes a conversation.
 704    /// </summary>
 705    /// <param name="conversationId">The conversation identifier.</param>
 706    /// <returns>True if removed, false if not found.</returns>
 707    public bool Remove(string conversationId)
 708    {
 709        ObjectDisposedException.ThrowIf(_disposed, this);
 710        return _conversations.TryRemove(conversationId, out _);
 711    }
 712
 713    /// <summary>
 714    /// Clears all conversations.
 715    /// </summary>
 716    public void Clear()
 717    {
 718        ObjectDisposedException.ThrowIf(_disposed, this);
 719        _conversations.Clear();
 720    }
 721
 722    /// <summary>
 723    /// Gets the number of active conversations.
 724    /// </summary>
 725    public int Count => _conversations.Count;
 726
 727    private void CleanupExpired(object? state)
 728    {
 729        if (_disposed) return;
 730
 731        var now = DateTime.UtcNow;
 732        var expired = _conversations
 733            .Where(kvp => now - kvp.Value.LastAccessedAt > _ttl)
 734            .Select(kvp => kvp.Key)
 735            .ToList();
 736
 737        foreach (var id in expired)
 738        {
 739            _conversations.TryRemove(id, out _);
 740        }
 741    }
 742
 743    /// <inheritdoc/>
 744    public void Dispose()
 745    {
 746        if (_disposed) return;
 747        _disposed = true;
 748        _cleanupTimer.Dispose();
 749        _conversations.Clear();
 750    }
 751}

 827    /// <summary>
 828    /// Builds the full message list including history.
 829    /// Trims to fit within token budget using a simple heuristic.
 830    /// </summary>
 831    /// <param name="newMessages">New messages to append.</param>
 832    /// <param name="maxTokens">Maximum tokens for history (rough estimate: 4 chars = 1 token).</param>
 833    /// <returns>Combined message list with history.</returns>
 834    public List<ChatMessage> BuildMessages(List<ChatMessage> newMessages, int maxTokens)
 835    {
 836        lock (_lock)
 837        {
 838            var combined = new List<ChatMessage>(_messages);
 839            combined.AddRange(newMessages);
 840
 841            // Simple token estimation: 4 characters  1 token
 842            var estimatedTokens = combined.Sum(m => (m.Content?.Length ?? 0) / 4);
 843
 844            if (estimatedTokens <= maxTokens)
 845            {
 846                return combined;
 847            }
 848
 849            // Trim older messages (keep system and most recent)
 850            var systemMessages = combined.Where(m => m.Role == "system").ToList();
 851            var otherMessages = combined.Where(m => m.Role != "system").ToList();
 852
 853            var result = new List<ChatMessage>(systemMessages);
 854            var currentTokens = systemMessages.Sum(m => (m.Content?.Length ?? 0) / 4);
 855
 856            // Add messages from most recent until we hit limit
 857            for (int i = otherMessages.Count - 1; i >= 0; i--)
 858            {
 859                var msgTokens = (otherMessages[i].Content?.Length ?? 0) / 4;
 860                if (currentTokens + msgTokens > maxTokens)
 861                {
 862                    break;
 863                }
 864                result.Insert(systemMessages.Count, otherMessages[i]);
 865                currentTokens += msgTokens;
 866            }
 867
 868            return result;
 869        }
 870    }

 872    /// <summary>
 873    /// Clears all messages from the conversation.
 874    /// </summary>
 875    public void Clear()

 880            Touch();

 889        LastAccessedAt = DateTime.UtcNow;
 890    }
 891}
