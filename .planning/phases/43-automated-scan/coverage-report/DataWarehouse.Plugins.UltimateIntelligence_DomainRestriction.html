 968/// <summary>
 969/// Sliding window counter for rate limiting.
 970/// </summary>
 971internal sealed class SlidingWindowCounter
 972{
 973    private readonly int _windowSeconds;
 974    private readonly ConcurrentQueue<DateTime> _timestamps = new();
 975    private readonly object _lockObject = new();
 976
 977    public SlidingWindowCounter(int windowSeconds)
 978    {
 979        _windowSeconds = windowSeconds;
 980    }
 981
 982    public void Increment()
 983    {
 984        CleanupOldEntries();
 985        _timestamps.Enqueue(DateTime.UtcNow);
 986    }
 987
 988    public long GetCount()
 989    {
 990        CleanupOldEntries();
 991        return _timestamps.Count;
 992    }
 993
 994    public TimeSpan GetTimeUntilReset()
 995    {
 996        if (_timestamps.TryPeek(out var oldest))
 997        {
 998            var resetTime = oldest.AddSeconds(_windowSeconds);
 999            var remaining = resetTime - DateTime.UtcNow;
 1000            return remaining > TimeSpan.Zero ? remaining : TimeSpan.Zero;
 1001        }
 1002
 1003        return TimeSpan.Zero;
 1004    }
 1005
 1006    private void CleanupOldEntries()
 1007    {
 1008        var threshold = DateTime.UtcNow.AddSeconds(-_windowSeconds);
 1009
 1010        lock (_lockObject)
 1011        {
 1012            while (_timestamps.TryPeek(out var oldest) && oldest < threshold)
 1013            {
 1014                _timestamps.TryDequeue(out _);
 1015            }
 1016        }
 1017    }
 1018}
