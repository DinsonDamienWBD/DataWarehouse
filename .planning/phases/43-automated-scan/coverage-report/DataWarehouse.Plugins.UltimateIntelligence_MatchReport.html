 230    private static double JaccardSimilarity(HashSet<string> setA, HashSet<string> setB)
 231    {
 232        if (setA.Count == 0 && setB.Count == 0)
 233            return 1.0;

 236        var union = setA.Union(setB).Count();

 414        var totalDocs = Volatile.Read(ref _totalDocuments);
 415        var allTerms = new HashSet<string>(termFreqs1.Keys, StringComparer.OrdinalIgnoreCase);
 416        allTerms.UnionWith(termFreqs2.Keys);
 417
 418        var totalTerms1 = termFreqs1.Values.Sum();
 419        var totalTerms2 = termFreqs2.Values.Sum();
 420
 421        if (totalTerms1 == 0 || totalTerms2 == 0)
 422            return 0.0;
 423
 424        var dotProduct = 0.0;
 425        var magnitude1 = 0.0;
 426        var magnitude2 = 0.0;
 427
 428        foreach (var term in allTerms)
 429        {
 430            var docFreq = _documentFreqs.TryGetValue(term, out var df) ? df : 0;
 431            var idf = Math.Log((double)totalDocs / (1 + docFreq));
 432
 433            var tf1 = termFreqs1.TryGetValue(term, out var freq1) ? (double)freq1 / totalTerms1 : 0.0;
 434            var tf2 = termFreqs2.TryGetValue(term, out var freq2) ? (double)freq2 / totalTerms2 : 0.0;
 435
 436            var tfidf1 = tf1 * idf;
 437            var tfidf2 = tf2 * idf;
 438
 439            dotProduct += tfidf1 * tfidf2;
 440            magnitude1 += tfidf1 * tfidf1;
 441            magnitude2 += tfidf2 * tfidf2;
 442        }
 443
 444        var mag1 = Math.Sqrt(magnitude1);
 445        var mag2 = Math.Sqrt(magnitude2);
 446
 447        if (mag1 == 0 || mag2 == 0)
 448            return 0.0;
 449
 450        return dotProduct / (mag1 * mag2);
 451    }

 453    private static string[] Tokenize(string content)
 454    {
 455        return Regex.Split(content, @"[^a-zA-Z0-9]+")
 456            .Where(t => t.Length > 0)
 457            .Select(t => t.ToLowerInvariant())
 458            .ToArray();
 459    }

 717        Description = "Matches semantically equivalent fields across different systems using name normalization, type co
 718        Capabilities = IntelligenceCapabilities.SemanticSearch | IntelligenceCapabilities.Classification,
 719        CostTier = 3,
 720        LatencyTier = 3,

 743    /// <summary>
 744    /// Finds semantically matching fields between two registered systems by comparing
 745    /// normalized field names, data types, and sample value overlap.
 746    /// </summary>
 747    /// <param name="system1Id">First system identifier.</param>
 748    /// <param name="system2Id">Second system identifier.</param>
 749    /// <param name="minConfidence">Minimum confidence threshold for a match to be included (0.0 to 1.0).</param>
 750    /// <returns>A <see cref="MatchReport"/> containing all discovered field matches.</returns>
 751    /// <exception cref="InvalidOperationException">If either system has not been registered.</exception>
 752    public MatchReport FindMatches(string system1Id, string system2Id, double minConfidence = 0.5)
 753    {
 754        if (!_systems.TryGetValue(system1Id, out var system1))
 755            throw new InvalidOperationException($"System '{system1Id}' has not been registered.");
 756        if (!_systems.TryGetValue(system2Id, out var system2))
 757            throw new InvalidOperationException($"System '{system2Id}' has not been registered.");
 758
 759        var matches = new List<SemanticFieldMatch>();
 760        var totalFieldsCompared = 0;
 761
 762        foreach (var (field1Name, field1Info) in system1.Fields)
 763        {
 764            var norm1 = NormalizeName(field1Name);
 765
 766            foreach (var (field2Name, field2Info) in system2.Fields)
 767            {
 768                totalFieldsCompared++;
 769                var norm2 = NormalizeName(field2Name);
 770
 771                var confidence = 0.0;
 772                var matchReason = "";
 773
 774                // 1. Exact normalized match
 775                if (string.Equals(norm1, norm2, StringComparison.OrdinalIgnoreCase))
 776                {
 777                    confidence = 1.0;
 778                    matchReason = "exact_name_match";
 779                }
 780                // 2. One name contains the other
 781                else if (norm1.Contains(norm2, StringComparison.OrdinalIgnoreCase) ||
 782                         norm2.Contains(norm1, StringComparison.OrdinalIgnoreCase))
 783                {
 784                    confidence = 0.8;
 785                    matchReason = "partial_name_match";
 786                }
 787                // 3. Same data type + similar name (shared character ratio > 0.6)
 788                else if (string.Equals(field1Info.DataType, field2Info.DataType, StringComparison.OrdinalIgnoreCase) &&
 789                         SharedCharRatio(norm1, norm2) > 0.6)
 790                {
 791                    confidence = 0.6;
 792                    matchReason = "type_and_name_similarity";
 793                }
 794
 795                // 4. Sample value overlap bonus
 796                if (confidence > 0 &&
 797                    field1Info.SampleValues is { Length: > 0 } &&
 798                    field2Info.SampleValues is { Length: > 0 })
 799                {
 800                    var set1 = new HashSet<string>(field1Info.SampleValues, StringComparer.OrdinalIgnoreCase);
 801                    var set2 = new HashSet<string>(field2Info.SampleValues, StringComparer.OrdinalIgnoreCase);
 802                    var intersection = set1.Intersect(set2).Count();
 803                    var union = set1.Union(set2).Count();
 804
 805                    if (union > 0 && (double)intersection / union > 0.3)
 806                    {
 807                        confidence += 0.2;
 808                        matchReason += "+value_overlap";
 809                    }
 810                }
 811
 812                if (confidence >= minConfidence)
 813                {
 814                    matches.Add(new SemanticFieldMatch(
 815                        System1Id: system1Id,
 816                        Field1Name: field1Name,
 817                        System2Id: system2Id,
 818                        Field2Name: field2Name,
 819                        Confidence: confidence,
 820                        MatchReason: matchReason));
 821                }
 822            }
 823        }
 824
 825        return new MatchReport(
 826            Matches: matches,
 827            TotalFieldsCompared: totalFieldsCompared,
 828            MatchesFound: matches.Count);
 829    }

 831    private static string NormalizeName(string name)
 832    {
 833        var normalized = name.ToLowerInvariant();
 834
 835        // Strip common prefixes
 836        foreach (var prefix in CommonPrefixes)
 837        {
 838            if (normalized.StartsWith(prefix, StringComparison.Ordinal))
 839            {
 840                normalized = normalized[prefix.Length..];
 841                break;
 842            }
 843        }
 844
 845        // Remove underscores and hyphens
 846        normalized = normalized.Replace("_", "").Replace("-", "");
 847
 848        return normalized;
 849    }

 851    private static double SharedCharRatio(string a, string b)
 852    {
 853        if (a.Length == 0 && b.Length == 0) return 1.0;
 854        if (a.Length == 0 || b.Length == 0) return 0.0;
 855
 856        // Count character frequencies
 857        var freqA = new Dictionary<char, int>();
 858        foreach (var c in a.ToLowerInvariant())
 859        {
 860            freqA[c] = freqA.GetValueOrDefault(c) + 1;
 861        }
 862
 863        var freqB = new Dictionary<char, int>();
 864        foreach (var c in b.ToLowerInvariant())
 865        {
 866            freqB[c] = freqB.GetValueOrDefault(c) + 1;
 867        }
 868
 869        // Count shared characters (minimum of each character's frequency)
 870        var sharedCount = 0;
