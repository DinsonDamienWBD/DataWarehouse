 568
 569        /// <summary>
 570        /// Executes a probabilistic query with SQL-like syntax (T85.8).
 571        /// </summary>
 572        /// <param name="query">Query string (e.g., "SELECT COUNT_DISTINCT(user_id) FROM events")</param>
 573        /// <returns>Query result with accuracy metadata.</returns>
 574        public ProbabilisticQueryResult ExecuteQuery(string query)
 575        {
 576            Interlocked.Increment(ref _totalQueriesServed);
 577
 578            // Parse simple query syntax
 579            var parts = query.Split(' ', StringSplitOptions.RemoveEmptyEntries);
 580            if (parts.Length < 4 || !parts[0].Equals("SELECT", StringComparison.OrdinalIgnoreCase))
 581            {
 582                return new ProbabilisticQueryResult
 583                {
 584                    Success = false,
 585                    Error = "Invalid query syntax. Expected: SELECT <function>(<field>) FROM <structure>"
 586                };
 587            }
 588
 589            var function = parts[1].ToUpperInvariant();
 590            var from = Array.IndexOf(parts.Select(p => p.ToUpperInvariant()).ToArray(), "FROM");
 591            if (from < 0 || from >= parts.Length - 1)
 592            {
 593                return new ProbabilisticQueryResult
 594                {
 595                    Success = false,
 596                    Error = "Missing FROM clause."
 597                };
 598            }
 599
 600            var structureName = parts[from + 1];
 601
 602            // Handle different query types
 603            if (function.StartsWith("COUNT_DISTINCT"))
 604            {
 605                var result = EstimateCardinality(structureName);
 606                Interlocked.Increment(ref _approximateQueriesServed);
 607                return new ProbabilisticQueryResult
 608                {
 609                    Success = true,
 610                    Value = result.Value,
 611                    Confidence = result.Confidence,
 612                    LowerBound = result.LowerBound,
 613                    UpperBound = result.UpperBound,
 614                    StructureType = "HyperLogLog"
 615                };
 616            }
 617            else if (function.StartsWith("COUNT("))
 618            {
 619                var item = ExtractFunctionArg(function);
 620                var result = EstimateFrequency(structureName, item);
 621                Interlocked.Increment(ref _approximateQueriesServed);
 622                return new ProbabilisticQueryResult
 623                {
 624                    Success = true,
 625                    Value = result.Value,
 626                    Confidence = result.Confidence,
 627                    LowerBound = result.LowerBound,
 628                    UpperBound = result.UpperBound,
 629                    StructureType = "CountMinSketch"
 630                };
 631            }
 632            else if (function.StartsWith("PERCENTILE"))
 633            {
 634                var pct = double.Parse(ExtractFunctionArg(function)) / 100.0;
 635                var result = EstimateQuantile(structureName, pct);
 636                Interlocked.Increment(ref _approximateQueriesServed);
 637                return new ProbabilisticQueryResult
 638                {
 639                    Success = true,
 640                    ValueDouble = result.Value,
 641                    Confidence = result.Confidence,
 642                    LowerBound = result.LowerBound,
 643                    UpperBound = result.UpperBound,
 644                    StructureType = "TDigest"
 645                };
 646            }
 647            else if (function.StartsWith("EXISTS("))
 648            {
 649                var item = ExtractFunctionArg(function);
 650                var result = TestBloomFilter(structureName, item);
 651                Interlocked.Increment(ref _approximateQueriesServed);
 652                return new ProbabilisticQueryResult
 653                {
 654                    Success = true,
 655                    BoolValue = result.Value,
 656                    Confidence = result.Confidence,
 657                    MayBeFalsePositive = result.MayBeFalsePositive,
 658                    StructureType = "BloomFilter"
 659                };
 660            }
 661
 662            return new ProbabilisticQueryResult
 663            {
 664                Success = false,
 665                Error = $"Unknown function: {function}"
 666            };
 667        }

 669        private string ExtractFunctionArg(string function)
 670        {
 671            var start = function.IndexOf('(') + 1;

 673            if (start <= 0 || end <= start) return string.Empty;
 674            return function.Substring(start, end - start).Trim('\'', '"');
 675        }

 745        /// <summary>
 746        /// Compares probabilistic vs exact results for validation (T85.10).
 747        /// </summary>
 748        public AccuracyReport CompareAccuracy(string name)
 749        {
 750            BloomFilterAccuracy? bloomAccuracy = null;
 751            CardinalityAccuracy? hllAccuracy = null;
 752
 753            // Compare Bloom filter
 754            if (_bloomFilters.TryGetValue(name, out var bloom) && _exactSets.TryGetValue(name, out var exactSet))
 755            {
 756                var falsePositives = 0;
 757                var truePositives = 0;
 758                var trueNegatives = 0;
 759
 760                // Test sample of items
 761                foreach (var item in exactSet)
 762                {
 763                    if (bloom.MightContain(item))
 764                        truePositives++;
 765                }
 766
 767                // Test some items not in set
 768                for (int i = 0; i < exactSet.Count; i++)
 769                {
 770                    var testItem = $"_test_not_in_set_{i}_{Guid.NewGuid()}";
 771                    if (exactSet.Contains(testItem)) continue;
 772
 773                    if (bloom.MightContain(testItem))
 774                        falsePositives++;
 775                    else
 776                        trueNegatives++;
 777                }
 778
 779                bloomAccuracy = new BloomFilterAccuracy
 780                {
 781                    TruePositives = truePositives,
 782                    FalsePositives = falsePositives,
 783                    TrueNegatives = trueNegatives,
 784                    ActualFPR = trueNegatives + falsePositives > 0
 785                        ? (double)falsePositives / (trueNegatives + falsePositives)
 786                        : 0,
 787                    ConfiguredFPR = bloom.ConfiguredErrorRate
 788                };
 789            }
 790
 791            // Compare HyperLogLog
 792            if (_hyperLogLogs.TryGetValue(name, out var hll) && _exactSets.TryGetValue(name, out var exactSet2))
 793            {
 794                var estimate = hll.Count();
 795                var exact = exactSet2.Count;
 796                hllAccuracy = new CardinalityAccuracy
 797                {
 798                    Estimate = estimate,
 799                    Exact = exact,
 800                    RelativeError = exact > 0 ? Math.Abs(estimate - exact) / (double)exact : 0,
 801                    StandardError = hll.StandardError
 802                };
 803            }
 804
 805            return new AccuracyReport
 806            {
 807                StructureName = name,
 808                BloomFilterAccuracy = bloomAccuracy,
 809                HyperLogLogAccuracy = hllAccuracy
 810            };
 811        }
