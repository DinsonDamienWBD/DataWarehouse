 259    {

 625    {

 627        _checkpointInterval = TimeSpan.FromMinutes(checkpointIntervalMinutes);
 628        _checkpointDirectory = Path.Combine(Path.GetTempPath(), "automl-checkpoints");
 629        Directory.CreateDirectory(_checkpointDirectory);

 631    }

 633    /// <summary>
 634    /// Saves a training checkpoint if the interval has elapsed.
 635    /// </summary>
 636    /// <param name="checkpoint">Checkpoint data containing model state.</param>
 637    /// <param name="ct">Cancellation token.</param>
 638    /// <exception cref="CheckpointException">If checkpoint save fails.</exception>
 639    public async Task SaveCheckpointAsync(TrainingCheckpoint checkpoint, CancellationToken ct = default)
 640    {
 641        try
 642        {
 643            var now = DateTime.UtcNow;
 644            var key = checkpoint.ModelId;
 645
 646            // Check if we should save based on interval
 647            if (_lastCheckpoints.TryGetValue(key, out var lastTime))
 648            {
 649                if (now - lastTime < _checkpointInterval)
 650                    return; // Too soon, skip
 651            }
 652
 653            // Prepare checkpoint file
 654            var checkpointFile = Path.Combine(_checkpointDirectory, $"{key}_epoch{checkpoint.Epoch}.ckpt");
 655            var checkpointData = SerializeCheckpoint(checkpoint);
 656            await File.WriteAllBytesAsync(checkpointFile, checkpointData, ct);
 657
 658            // Persist to storage via message bus
 659            await PersistToStorageAsync(checkpointFile, ct);
 660
 661            _lastCheckpoints[key] = now;
 662        }
 663        catch (Exception ex) when (ex is not CheckpointException)
 664        {
 665            throw new CheckpointException($"Failed to save checkpoint: {ex.Message}", ex);
 666        }
 667    }

 829/// <summary>

 832/// </summary>
 833public sealed class EdgeResourceAwareTrainer
 834{
 835    private readonly IMessageBus? _messageBus;
 836    private readonly TimeSpan _metricsCheckInterval;
 837
 838    /// <summary>
 839    /// Initializes the resource-aware trainer.

 846        _metricsCheckInterval = TimeSpan.FromSeconds(metricsCheckIntervalSeconds);

 851    /// </summary>
 852    /// <param name="compiledPath">Path to the compiled training script/executable.</param>
 853    /// <param name="dataPath">Path to the training data.</param>
 854    /// <param name="checkpointCallback">Callback for saving checkpoints.</param>
 855    /// <param name="ct">Cancellation token.</param>
 856    /// <returns>Path to the trained model.</returns>
 857    /// <exception cref="TrainingException">If training fails.</exception>
 858    public async Task<string> TrainWithResourceMonitoringAsync(
 859        string compiledPath,
 860        string dataPath,
 861        Func<TrainingCheckpoint, Task>? checkpointCallback = null,
 862        CancellationToken ct = default)
 863    {

 869
 870            while (!trainingProcess.HasExited && !ct.IsCancellationRequested)
 871            {

 873

 880                    ThrottleProcess(trainingProcess);
 881                }

 894                        Timestamp = DateTime.UtcNow,
 895                        ModelState = new Dictionary<string, object>()
 896                    };
 897
 898                    await checkpointCallback(checkpoint);
 899                }
 900            }
 901
 902            await trainingProcess.WaitForExitAsync(ct);
 903
 904            if (trainingProcess.ExitCode != 0)
 905            {
 906                throw new TrainingException($"Training process exited with code {trainingProcess.ExitCode}");
 907            }
 908
 909            return outputPath;
 910        }
 911        catch (Exception ex) when (ex is not TrainingException)
 912        {
 913            throw new TrainingException($"Resource-aware training failed: {ex.Message}", ex);
 914        }
 915    }
