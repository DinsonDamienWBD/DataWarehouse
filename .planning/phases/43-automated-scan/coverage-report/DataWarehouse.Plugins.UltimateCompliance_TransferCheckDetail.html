 535        private TransferMechanism? FindApplicableMechanism(TransferCheckInput input)
 536        {
 537            foreach (var mechanism in _mechanisms.Values.Where(m => m.IsActive))
 538            {
 539                // Check if mechanism applies to this destination
 540                if (mechanism.ApplicableCountries != null &&
 541                    !mechanism.ApplicableCountries.Contains(input.DestinationCountry, StringComparer.OrdinalIgnoreCase))
 542                {
 543                    continue;
 544                }
 545
 546                // Check validity
 547                if (mechanism.ValidUntil.HasValue && mechanism.ValidUntil.Value < DateTime.UtcNow)
 548                {
 549                    continue;
 550                }
 551
 552                return mechanism;
 553            }
 554
 555            return null;
 556        }
