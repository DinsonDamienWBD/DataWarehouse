 805        /// <summary>
 806        /// Calculates the Shannon entropy of data (0-8 bits per byte).
 807        /// Higher entropy indicates more randomness (compressed/encrypted data).
 808        /// </summary>
 809        /// <param name="data">The data to analyze.</param>
 810        /// <returns>Entropy value (0-8).</returns>
 811        protected static double CalculateEntropy(ReadOnlySpan<byte> data)
 812        {
 813            if (data.Length == 0)
 814                return 0;
 815
 816            Span<int> frequency = stackalloc int[256];
 817            frequency.Clear();
 818
 819            foreach (var b in data)
 820                frequency[b]++;
 821
 822            double entropy = 0.0;
 823            double length = data.Length;
 824
 825            for (int i = 0; i < 256; i++)
 826            {
 827                if (frequency[i] == 0)
 828                    continue;
 829
 830                double probability = frequency[i] / length;
 831                entropy -= probability * Math.Log2(probability);
 832            }
 833
 834            return entropy;
 835        }

 837        /// <summary>
 838        /// Checks if data starts with known compressed file signatures.
 839        /// </summary>
 840        private static bool IsCompressedFormat(ReadOnlySpan<byte> data)
 841        {
 842            if (data.Length < 4)
 843                return false;
 844
 845            // GZip: 1F 8B
 846            if (data[0] == 0x1F && data[1] == 0x8B)
 847                return true;
 848
 849            // ZIP: 50 4B 03 04 or 50 4B 05 06
 850            if (data[0] == 0x50 && data[1] == 0x4B && (data[2] == 0x03 || data[2] == 0x05))
 851                return true;
 852
 853            // Zstd: 28 B5 2F FD
 854            if (data[0] == 0x28 && data[1] == 0xB5 && data[2] == 0x2F && data[3] == 0xFD)
 855                return true;
 856
 857            // Brotli has no magic number, check via other means
 858            return false;
 859        }

 861        /// <summary>
 862        /// Checks if data starts with known media file signatures.
 863        /// </summary>
 864        private static bool IsMediaFormat(ReadOnlySpan<byte> data)
 865        {
 866            if (data.Length < 4)
 867                return false;
 868
 869            // JPEG: FF D8 FF
 870            if (data[0] == 0xFF && data[1] == 0xD8 && data[2] == 0xFF)
 871                return true;
 872
 873            // PNG: 89 50 4E 47
 874            if (data[0] == 0x89 && data[1] == 0x50 && data[2] == 0x4E && data[3] == 0x47)
 875                return true;
 876
 877            // MP3: ID3 or FF FB/FF F3
 878            if ((data[0] == 0x49 && data[1] == 0x44 && data[2] == 0x33) ||
 879                (data[0] == 0xFF && (data[1] == 0xFB || data[1] == 0xF3)))
 880                return true;
 881
 882            // MP4/MOV: ftyp
 883            if (data.Length >= 8 && data[4] == 0x66 && data[5] == 0x74 && data[6] == 0x79 && data[7] == 0x70)
 884                return true;
 885
 886            return false;
 887        }

 889        /// <summary>
 890        /// Checks if data appears to be text (high proportion of printable characters).
 891        /// </summary>
 892        private static bool IsTextFormat(ReadOnlySpan<byte> data)
 893        {
 894            if (data.Length == 0)
 895                return false;
 896
 897            int printableCount = 0;
 898            int sampleSize = Math.Min(data.Length, 512);
 899
 900            for (int i = 0; i < sampleSize; i++)
 901            {
 902                byte b = data[i];
 903                // Printable ASCII + common whitespace
 904                if ((b >= 32 && b <= 126) || b == 9 || b == 10 || b == 13)
 905                    printableCount++;
 906            }
 907
 908            // If > 90% printable, likely text
 909            return (double)printableCount / sampleSize > 0.90;
 910        }
