 559            {

 565            }

 570                return new SessionValidationResult

 585                    ReauthenticationReason = "IdP token expired"
 586                };
 587            }
 588
 589            // Update last activity
 590            _sessions[sessionId] = session with { LastActivityAt = DateTime.UtcNow };
 591
 592            return new SessionValidationResult
 593            {
 594                IsValid = true,
 595                Session = session
 596            };
 597        }

 777                .OrderBy(t => t.Priority);

 787        private List<FederatedClaim> ApplyTransformation(List<FederatedClaim> claims, ClaimsTransformation transformatio
 788        {
 789            var result = new List<FederatedClaim>();
 790
 791            foreach (var claim in claims)
 792            {
 793                // Check for rename
 794                if (transformation.RenameRules.TryGetValue(claim.Type, out var newType))
 795                {
 796                    result.Add(claim with { Type = newType });
 797                    continue;
 798                }
 799
 800                // Check for filter (remove)
 801                if (transformation.FilterRules.Contains(claim.Type))
 802                {
 803                    continue;
 804                }
 805
 806                // Check for value transformation
 807                if (transformation.ValueTransformRules.TryGetValue(claim.Type, out var transformer))
 808                {
 809                    result.Add(claim with { Value = transformer(claim.Value) });
 810                    continue;
 811                }
 812
 813                result.Add(claim);
 814            }
 815
 816            // Add synthesized claims
 817            foreach (var synth in transformation.SynthesizeRules)
 818            {
 819                var value = synth.ValueGenerator(claims);
 820                if (!string.IsNullOrEmpty(value))
 821                {
 822                    result.Add(new FederatedClaim
 823                    {
 824                        Type = synth.ClaimType,
 825                        Value = value,
 826                        Issuer = "ClaimsTransformation"
 827                    });
 828                }
 829            }
 830
 831            return result;
 832        }

 858        /// <inheritdoc/>
 859        protected override async Task<AccessDecision> EvaluateAccessCoreAsync(AccessContext context, CancellationToken c
 860        {

 863                tokenObj is not string token)

 904                            ["IdpId"] = sessionResult.Session!.IdpId,

 906                        }
 907                    };
 908                }

 910                return new AccessDecision
 911                {
 912                    IsGranted = false,
 913                    Reason = "No federation token or session provided",

 915                };
 916            }
 917
 918            // Get IdP ID from context
 919            if (!context.EnvironmentAttributes.TryGetValue("IdpId", out var idpIdObj) ||
 920                idpIdObj is not string idpId)
 921            {
 922                return new AccessDecision
 923                {

 925                    Reason = "Identity provider not specified",

 938                    Reason = validationResult.Error ?? "Token validation failed",

 940                    Metadata = new Dictionary<string, object>
 941                    {

 943                    }
 944                };
 945            }
 946
 947            // Resolve or provision user

 951            if (mapping == null)
 952            {
 953                if (!_enableJitProvisioning)
 954                {
 955                    return new AccessDecision
 956                    {
 957                        IsGranted = false,
 958                        Reason = "User not mapped and JIT provisioning is disabled",

 969                        Reason = provisionResult.Error ?? "User provisioning failed",
 970                        ApplicablePolicies = new[] { "FederatedIdentity.ProvisioningFailed" }

 972                }
 973
 974                localUserId = provisionResult.LocalUserId!;
 975            }

 977            {
 978                localUserId = mapping.LocalUserId;
 979            }
 980
 981            // Create or update session
 982            var session = CreateSession(localUserId, idpId, validationResult);
 983
 984            return new AccessDecision
 985            {
 986                IsGranted = true,
 987                Reason = "Federated authentication successful",
 988                ApplicablePolicies = new[] { "FederatedIdentity.Authenticated" },
 989                Metadata = new Dictionary<string, object>
 990                {
 991                    ["SessionId"] = session.Id,
 992                    ["IdpId"] = idpId,
 993                    ["LocalUserId"] = localUserId,
 994                    ["IdpSubjectId"] = validationResult.SubjectId!,
 995                    ["Claims"] = validationResult.Claims != null
 996                        ? validationResult.Claims.Select(c => $"{c.Type}={c.Value}").ToList()
 997                        : new List<string>()
 998                }
 999            };
 1000        }
