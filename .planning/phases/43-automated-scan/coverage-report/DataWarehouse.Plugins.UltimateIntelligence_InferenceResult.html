 551/// <summary>
 552/// LRU cache for loaded ML models with memory pressure awareness.
 553/// Prevents redundant model loading and optimizes inference performance.
 554/// </summary>
 555public class WasiNnModelCache
 556{
 557    private readonly int _maxCacheSize;
 558    private readonly long _maxMemoryBytes;
 559    private readonly ConcurrentDictionary<string, CachedModel> _cache = new();
 560    private readonly LinkedList<string> _lruList = new();
 561    private readonly object _lock = new();
 562    private long _currentMemoryUsage = 0;
 563
 564    /// <summary>
 565    /// Initializes a new instance of the WasiNnModelCache.
 566    /// </summary>
 567    /// <param name="maxCacheSize">Maximum number of models to cache.</param>
 568    /// <param name="maxMemoryMb">Maximum memory usage in MB.</param>
 569    public WasiNnModelCache(int maxCacheSize = 10, int maxMemoryMb = 512)
 570    {
 571        _maxCacheSize = maxCacheSize;
 572        _maxMemoryBytes = maxMemoryMb * 1024L * 1024L;
 573    }
 574
 575    /// <summary>
 576    /// Gets the current number of cached models.
 577    /// </summary>
 578    public int Count => _cache.Count;
 579
 580    /// <summary>
 581    /// Gets the current memory usage in bytes.
 582    /// </summary>
 583    public long CurrentMemoryUsage => Interlocked.Read(ref _currentMemoryUsage);
 584
 585    /// <summary>
 586    /// Gets or adds a model to the cache.
 587    /// </summary>
 588    /// <param name="modelPath">Path to the model file.</param>
 589    /// <param name="loader">Function to load the model if not cached.</param>
 590    /// <param name="ct">Cancellation token.</param>
 591    /// <returns>Cached model data.</returns>
 592    public async Task<byte[]> GetOrAddAsync(string modelPath, Func<Task<byte[]>> loader, CancellationToken ct = default)
 593    {
 594        // Check if already cached
 595        if (_cache.TryGetValue(modelPath, out var cached))
 596        {
 597            lock (_lock)
 598            {
 599                // Move to front of LRU list
 600                _lruList.Remove(cached.LruNode);
 601                _lruList.AddFirst(cached.LruNode);
 602            }
 603
 604            cached.LastAccessTime = DateTime.UtcNow;
 605            cached.HitCount++;
 606            return cached.Data;
 607        }
 608
 609        // Load model
 610        var data = await loader();
 611        var sizeBytes = data.Length;
 612
 613        lock (_lock)
 614        {
 615            // Evict if necessary
 616            while ((_cache.Count >= _maxCacheSize || _currentMemoryUsage + sizeBytes > _maxMemoryBytes) && _lruList.Coun
 617            {
 618                EvictLeastRecentlyUsed();
 619            }
 620
 621            // Add to cache
 622            var node = _lruList.AddFirst(modelPath);
 623            var model = new CachedModel
 624            {
 625                Path = modelPath,
 626                Data = data,
 627                SizeBytes = sizeBytes,
 628                LoadedAt = DateTime.UtcNow,
 629                LastAccessTime = DateTime.UtcNow,
 630                LruNode = node,
 631                HitCount = 0
 632            };
 633
 634            _cache[modelPath] = model;
 635            Interlocked.Add(ref _currentMemoryUsage, sizeBytes);
 636        }
 637
 638        return data;
 639    }

 641    /// <summary>
 642    /// Removes a model from the cache.
 643    /// </summary>
 644    /// <param name="modelPath">Path to the model file.</param>
 645    /// <returns>True if removed, false if not found.</returns>
 646    public bool Remove(string modelPath)
 647    {
 648        if (_cache.TryRemove(modelPath, out var cached))
 649        {
 650            lock (_lock)
 651            {
 652                _lruList.Remove(cached.LruNode);
 653                Interlocked.Add(ref _currentMemoryUsage, -cached.SizeBytes);
 654            }
 655            return true;
 656        }
 657        return false;
 658    }

 660    /// <summary>
 661    /// Clears all cached models.
 662    /// </summary>
 663    public void Clear()
 664    {
 665        lock (_lock)
 666        {
 667            _cache.Clear();
 668            _lruList.Clear();
 669            Interlocked.Exchange(ref _currentMemoryUsage, 0);
 670        }
 671    }

 673    /// <summary>
 674    /// Gets cache statistics.
 675    /// </summary>
 676    /// <returns>Cache statistics.</returns>
 677    public CacheStatistics GetStatistics()
 678    {
 679        var models = _cache.Values.ToArray();
 680        return new CacheStatistics
 681        {
 682            CachedModels = models.Length,
 683            TotalMemoryBytes = Interlocked.Read(ref _currentMemoryUsage),
 684            MaxMemoryBytes = _maxMemoryBytes,
 685            MaxCacheSize = _maxCacheSize,
 686            TotalHits = models.Sum(m => m.HitCount),
 687            AverageModelSizeBytes = models.Length > 0 ? models.Average(m => m.SizeBytes) : 0,
 688            OldestModelAge = models.Length > 0 ? models.Max(m => (DateTime.UtcNow - m.LoadedAt).TotalSeconds) : 0
 689        };
 690    }

 692    private void EvictLeastRecentlyUsed()
 693    {
 694        var lruPath = _lruList.Last?.Value;
 695        if (lruPath != null && _cache.TryRemove(lruPath, out var evicted))
 696        {
 697            _lruList.RemoveLast();
 698            Interlocked.Add(ref _currentMemoryUsage, -evicted.SizeBytes);
 699        }
 700    }

 934/// <summary>

 968        }
 969    }

 972    /// Gets the total number of inferences processed.
 973    /// </summary>
 974    public long TotalInferences => Interlocked.Read(ref _totalInferences);
 975
 976    /// <summary>
 977    /// Processes a stream of inference requests with zero-copy optimization.
 978    /// </summary>
 979    /// <param name="modelPath">Path to the model.</param>
 980    /// <param name="inputStream">Stream of input data.</param>
 981    /// <param name="options">Inference options.</param>
 982    /// <param name="ct">Cancellation token.</param>
 983    /// <returns>Stream of inference results.</returns>
 984    public async IAsyncEnumerable<InferenceResult> ProcessStreamAsync(
 985        string modelPath,
 986        IAsyncEnumerable<InferenceInput> inputStream,
 987        InferenceOptions? options = null,
 988        [System.Runtime.CompilerServices.EnumeratorCancellation] CancellationToken ct = default)
 989    {
 990        options ??= new InferenceOptions();
 991
 992        // Pre-warm cache
 993        if (options.UseCache)
 994        {
 995            await _modelCache.GetOrAddAsync(modelPath, () => File.ReadAllBytesAsync(modelPath), ct);
 996        }
 997
 998        await foreach (var input in inputStream.WithCancellation(ct))
 999        {
 1000            await _semaphore.WaitAsync(ct);
 1001
 1002            try
 1003            {
 1004                var sw = Stopwatch.StartNew();
 1005                var result = await _strategy.InferAsync(modelPath, input, options, ct);
 1006                sw.Stop();
 1007
 1008                Interlocked.Increment(ref _totalInferences);
 1009                Interlocked.Add(ref _totalLatencyMs, (long)sw.Elapsed.TotalMilliseconds);
 1010
 1011                yield return result;
 1012            }
 1013            finally
 1014            {
 1015                _semaphore.Release();
 1016            }
 1017        }
 1018    }
 1019
 1020    /// <summary>
 1021    /// Processes a batch of inputs in parallel.
 1022    /// </summary>
 1023    /// <param name="modelPath">Path to the model.</param>
 1024    /// <param name="inputs">Batch of inputs.</param>
 1025    /// <param name="options">Inference options.</param>
 1026    /// <param name="ct">Cancellation token.</param>
 1027    /// <returns>Array of inference results.</returns>
 1028    public async Task<InferenceResult[]> ProcessBatchAsync(
 1029        string modelPath,
 1030        InferenceInput[] inputs,
 1031        InferenceOptions? options = null,
 1032        CancellationToken ct = default)
 1033    {
 1034        var tasks = inputs.Select(input => ProcessSingleAsync(modelPath, input, options, ct));
 1035        return await Task.WhenAll(tasks);
 1036    }

 1038    private async Task<InferenceResult> ProcessSingleAsync(
 1039        string modelPath,
 1040        InferenceInput input,
 1041        InferenceOptions? options,
 1042        CancellationToken ct)
 1043    {
 1044        await _semaphore.WaitAsync(ct);
 1045
 1046        try
