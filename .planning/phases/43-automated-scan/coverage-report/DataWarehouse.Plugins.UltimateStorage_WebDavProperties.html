 979        /// <summary>
 980        /// Executes an HTTP request with retry logic and exponential backoff.
 981        /// </summary>
 982        private async Task<HttpResponseMessage> ExecuteWithRetryAsync(HttpRequestMessage request, CancellationToken ct)
 983        {
 984            var attempt = 0;
 985            Exception? lastException = null;
 986
 987            while (attempt < _maxRetries)
 988            {
 989                try
 990                {
 991                    // Clone request for retry (HttpRequestMessage can only be sent once)
 992                    var clonedRequest = await CloneHttpRequestAsync(request);
 993                    var response = await _httpClient!.SendAsync(clonedRequest, ct);
 994
 995                    // Don't retry on client errors (4xx) except 408, 429
 996                    if (response.IsSuccessStatusCode ||
 997                        ((int)response.StatusCode >= 400 && (int)response.StatusCode < 500 &&
 998                         response.StatusCode != HttpStatusCode.RequestTimeout &&
 999                         response.StatusCode != (HttpStatusCode)429))
 1000                    {
 1001                        return response;
 1002                    }
 1003
 1004                    lastException = new HttpRequestException($"HTTP {response.StatusCode}: {response.ReasonPhrase}");
 1005                }
 1006                catch (Exception ex) when (ex is HttpRequestException || ex is TaskCanceledException)
 1007                {
 1008                    lastException = ex;
 1009                }
 1010
 1011                attempt++;
 1012
 1013                if (attempt < _maxRetries)
