 493    {

 495
 496        switch (index.IndexId)
 497        {

 499                if (queryType.HasFlag(QueryType.Hierarchical))
 500                    score += 0.5f;

 502                    score += 0.2f;
 503                break;
 504
 505            case "index-semantic-cluster":
 506                if (queryType.HasFlag(QueryType.Semantic))
 507                    score += 0.5f;

 509                    score += 0.3f;
 510                break;
 511
