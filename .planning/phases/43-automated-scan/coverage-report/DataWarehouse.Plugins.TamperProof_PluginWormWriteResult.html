 441            if (data.Length % raidConfig.DataShards != 0)

 443                shardSize++; // Round up for last shard
 444            }
 445
 446            // Create data shards
 447            var shards = new List<byte[]>();
 448            for (int i = 0; i < raidConfig.DataShards; i++)
 449            {
 450                var start = i * shardSize;
 451                var length = Math.Min(shardSize, data.Length - start);
 452                var shard = new byte[length];
 453                Array.Copy(data, start, shard, 0, length);
 454                shards.Add(shard);
 455            }
 456
 457            // Create parity shards (simple XOR for now, would use Reed-Solomon in production)
 458            for (int i = 0; i < raidConfig.ParityShards; i++)
 459            {
 460                var parityShard = new byte[shardSize];
 461                for (int j = 0; j < raidConfig.DataShards; j++)
 462                {
 463                    for (int k = 0; k < Math.Min(shardSize, shards[j].Length); k++)
 464                    {
 465                        parityShard[k] ^= shards[j][k];
 466                    }
 467                }
 468                shards.Add(parityShard);
 469            }
 470
 471            // Compute shard hashes
 472            var shardHashes = await integrity.ComputeShardHashesAsync(shards, objectId, ct);
 473
 474            // Build RAID record
 475            var shardRecords = new List<ShardRecord>();
 476            for (int i = 0; i < shards.Count; i++)
 477            {
 478                shardRecords.Add(new ShardRecord
 479                {
 480                    ShardIndex = i,
 481                    IsParity = i >= raidConfig.DataShards,
 482                    ActualSize = shards[i].Length,
 483                    ContentHash = shardHashes[i].Hash.HashValue,
 484                    StorageLocation = $"shard_{objectId}_{i}.bin",
 485                    WrittenAt = DateTimeOffset.UtcNow
 486                });
 487            }
 488
 489            var raidRecord = new RaidRecord
 490            {
 491                DataShardCount = raidConfig.DataShards,
 492                ParityShardCount = raidConfig.ParityShards,
 493                ShardSize = shardSize,
 494                Shards = shardRecords
 495            };
 496
 497            logger.LogDebug("RAID sharding complete: {TotalShards} shards created", shards.Count);
 498
 499            return (shards, raidRecord);
 500        }
 501        catch (Exception ex)
 502        {
 503            logger.LogError(ex, "Failed to perform RAID sharding");
 504            throw new InvalidOperationException("RAID sharding failed", ex);
 505        }
 506    }

 590    /// <summary>
 591    /// Internal implementation of transactional write (after seal verification).
 592    /// </summary>

 594        Guid objectId,
 595        TamperProofManifest manifest,
 596        List<byte[]> shards,
 597        byte[] fullData,
 598        IStorageProvider dataStorage,
 599        IStorageProvider metadataStorage,

 624                "Blockchain",

 630

 632                objectId,
 633                dataTierResult,

 656                    logger,
 657                    ct);
 658            }
 659
 660            return TransactionResult.CreateFailure(

 662                $"Transaction failed: {ex.Message}",
 663                InstanceDegradationState.Corrupted,
 664                rollbackAttempted,
 665                rollbackResult,
 666                dataTierResult,
 667                metadataTierResult,
 668                wormTierResult,
 669                blockchainTierResult);
 670        }
 671    }

 777    /// <summary>
 778    /// Updates the version index for an object.
 779    /// </summary>
 780    private static async Task UpdateVersionIndexAsync(
 781        Guid objectId,
 782        int newVersion,
 783        IStorageProvider metadataStorage,
 784        ILogger logger,
 785        CancellationToken ct)
 786    {
 787        try
 788        {
 789            var indexUri = new Uri($"metadata://manifests/{objectId}_versions.json");
 790            VersionIndexRecord index;
 791
 792            // Try to load existing index
 793            try
 794            {
 795                using var existingStream = await metadataStorage.LoadAsync(indexUri);
 796                if (existingStream != null)
 797                {
 798                    var existingJson = await new StreamReader(existingStream).ReadToEndAsync(ct);
 799                    var existingIndex = System.Text.Json.JsonSerializer.Deserialize<VersionIndexRecord>(existingJson);
 800                    if (existingIndex != null)
 801                    {
 802                        // Update existing index
 803                        var versions = existingIndex.AvailableVersions?.ToList() ?? new List<int>();
 804                        if (!versions.Contains(newVersion))
 805                        {
 806                            versions.Add(newVersion);
 807                        }
 808                        versions.Sort();
 809
 810                        index = new VersionIndexRecord
 811                        {
 812                            ObjectId = objectId,
 813                            LatestVersion = Math.Max(existingIndex.LatestVersion, newVersion),
 814                            UpdatedAt = DateTimeOffset.UtcNow,
 815                            AvailableVersions = versions
 816                        };
 817                    }
 818                    else
 819                    {
 820                        index = CreateNewIndex(objectId, newVersion);
 821                    }
 822                }
 823                else
 824                {
 825                    index = CreateNewIndex(objectId, newVersion);
 826                }
 827            }
 828            catch
 829            {
 830                // Index doesn't exist, create new one
 831                index = CreateNewIndex(objectId, newVersion);
 832            }
 833
 834            // Write updated index
 835            var indexJson = System.Text.Json.JsonSerializer.Serialize(index);
 836            using var indexMs = new MemoryStream(System.Text.Encoding.UTF8.GetBytes(indexJson));
 837            await metadataStorage.SaveAsync(indexUri, indexMs);
 838
 839            logger.LogDebug("Updated version index for object {ObjectId}: latest version {Version}",
 840                objectId, index.LatestVersion);
 841        }
 842        catch (Exception ex)
 843        {
 844            // Non-critical failure, log and continue
 845            logger.LogDebug(ex, "Failed to update version index for object {ObjectId}", objectId);
 846        }
 847    }

 924    /// <summary>

 927    /// </summary>
 928    private static async Task<DataWarehouse.SDK.Contracts.TamperProof.RollbackResult> RollbackTransactionAsync(
 929        Guid objectId,
 930        TierWriteResult? dataTierResult,
 931        TierWriteResult? metadataTierResult,
 932        TierWriteResult? wormTierResult,
 933        IStorageProvider dataStorage,
 934        IStorageProvider metadataStorage,
 935        ILogger logger,
 936        CancellationToken ct)
 937    {
 938        logger.LogWarning("Rolling back transaction for object {ObjectId}", objectId);
 939
 940        var tierResults = new List<TierRollbackResult>();
 941        var orphanedRecords = new List<OrphanedWormRecord>();
