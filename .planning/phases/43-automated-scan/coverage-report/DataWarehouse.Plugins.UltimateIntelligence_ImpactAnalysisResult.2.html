 736/// <summary>
 737/// Provides rollback guarantees for simulations.
 738/// Ensures simulated changes never affect real state unless explicitly committed.
 739/// </summary>
 740public sealed class RollbackGuarantee
 741{
 742    private readonly ConcurrentDictionary<string, List<RollbackPoint>> _rollbackPoints = new();
 743    private readonly object _lock = new();
 744
 745    /// <summary>
 746    /// Creates a rollback point for a session.
 747    /// </summary>
 748    /// <param name="sessionId">Session identifier.</param>
 749    /// <param name="forkId">Fork identifier.</param>
 750    /// <param name="description">Description of the rollback point.</param>
 751    /// <returns>The created rollback point.</returns>
 752    public RollbackPoint CreateRollbackPoint(string sessionId, string forkId, string description)
 753    {
 754        var point = new RollbackPoint
 755        {
 756            SessionId = sessionId,
 757            ForkId = forkId,
 758            Description = description
 759        };
 760
 761        _rollbackPoints.AddOrUpdate(
 762            sessionId,
 763            _ => new List<RollbackPoint> { point },
 764            (_, existing) =>
 765            {
 766                lock (_lock)
 767                {
 768                    existing.Add(point);
 769                    return existing;
 770                }
 771            });
 772
 773        return point;
 774    }
 775
 776    /// <summary>
 777    /// Gets all rollback points for a session.
 778    /// </summary>
 779    /// <param name="sessionId">Session identifier.</param>
 780    /// <returns>List of rollback points.</returns>
 781    public IReadOnlyList<RollbackPoint> GetRollbackPoints(string sessionId)
 782    {
 783        return _rollbackPoints.TryGetValue(sessionId, out var points)
 784            ? points.AsReadOnly()
 785            : Array.Empty<RollbackPoint>();
 786    }
 787
 788    /// <summary>
 789    /// Validates that a session can be safely rolled back.
 790    /// </summary>
 791    /// <param name="sessionId">Session identifier.</param>
 792    /// <returns>Validation result.</returns>
 793    public RollbackValidation ValidateRollback(string sessionId)
 794    {
 795        var points = GetRollbackPoints(sessionId);
 796
 797        if (points.Count == 0)
 798        {
 799            return new RollbackValidation
 800            {
 801                CanRollback = false,
 802                Reason = "No rollback points found for session"
 803            };
 804        }
 805
 806        var invalidPoints = points.Where(p => p.IsInvalidated).ToList();
 807        if (invalidPoints.Count == points.Count)
 808        {
 809            return new RollbackValidation
 810            {
 811                CanRollback = false,
 812                Reason = "All rollback points have been invalidated"
 813            };
 814        }
 815
 816        return new RollbackValidation
 817        {
 818            CanRollback = true,
 819            AvailablePoints = points.Where(p => !p.IsInvalidated).ToList()
 820        };
 821    }
 822
 823    /// <summary>
 824    /// Clears all rollback points for a session.
 825    /// </summary>
 826    /// <param name="sessionId">Session identifier.</param>
 827    public void ClearSession(string sessionId)
 828    {
 829        _rollbackPoints.TryRemove(sessionId, out _);
 830    }

 882/// <summary>
 883/// Simulates hypothetical changes in isolation.
 884/// </summary>
 885public sealed class ChangeSimulator
 886{
 887    private readonly ConcurrentDictionary<string, SimulatedChange> _pendingChanges = new();
 888
 889    /// <summary>
 890    /// Queues a change for simulation.
 891    /// </summary>
 892    /// <param name="change">Change to queue.</param>
 893    /// <returns>The queued change.</returns>
 894    public SimulatedChange QueueChange(SimulatedChange change)
 895    {
 896        _pendingChanges[change.ChangeId] = change;
 897        return change;
 898    }
 899
 900    /// <summary>
 901    /// Applies all pending changes to a fork.
 902    /// </summary>
 903    /// <param name="fork">Fork to apply changes to.</param>
 904    /// <param name="ct">Cancellation token.</param>
 905    /// <returns>Results of applying changes.</returns>
 906    public async Task<ChangeApplicationResult> ApplyPendingChangesAsync(
 907        StateFork fork,
 908        CancellationToken ct = default)
 909    {
 910        var results = new List<ChangeResult>();
 911        var changes = _pendingChanges.Values.OrderBy(c => c.AppliedAt).ToList();
 912
 913        foreach (var change in changes)
 914        {
 915            ct.ThrowIfCancellationRequested();
 916
 917            try
 918            {
 919                fork.ApplyChange(change);
 920                results.Add(new ChangeResult
 921                {
 922                    ChangeId = change.ChangeId,
 923                    Success = true,
 924                    AppliedAt = DateTime.UtcNow
 925                });
 926                _pendingChanges.TryRemove(change.ChangeId, out _);
 927            }
 928            catch (Exception ex)
 929            {
 930                results.Add(new ChangeResult
 931                {
 932                    ChangeId = change.ChangeId,
 933                    Success = false,
 934                    Error = ex.Message
 935                });
 936            }
 937        }
 938
 939        await Task.CompletedTask;
 940        return new ChangeApplicationResult
 941        {
 942            TotalChanges = changes.Count,
 943            SuccessfulChanges = results.Count(r => r.Success),
 944            FailedChanges = results.Count(r => !r.Success),
 945            Results = results
 946        };
 947    }
 948
 949    /// <summary>
 950    /// Validates changes before application.
 951    /// </summary>
 952    /// <param name="fork">Fork to validate against.</param>
 953    /// <param name="ct">Cancellation token.</param>
 954    /// <returns>Validation results.</returns>
 955    public async Task<ChangeValidationResult> ValidateChangesAsync(
 956        StateFork fork,
 957        CancellationToken ct = default)
 958    {
 959        var issues = new List<string>();
 960        var warnings = new List<string>();
 961
 962        foreach (var change in _pendingChanges.Values)
 963        {
 964            ct.ThrowIfCancellationRequested();
 965
 966            // Validate based on change type
 967            switch (change.Type)
 968            {
 969                case ChangeType.Update:
 970                case ChangeType.Delete:
 971                    if (!fork.StateSnapshot.ContainsKey(change.TargetPath))
 972                        warnings.Add($"Target path '{change.TargetPath}' does not exist for {change.Type} operation");
 973                    break;
 974
 975                case ChangeType.Create:
 976                    if (fork.StateSnapshot.ContainsKey(change.TargetPath))
 977                        issues.Add($"Target path '{change.TargetPath}' already exists for Create operation");
 978                    break;
 979            }
 980        }
 981
 982        await Task.CompletedTask;
 983        return new ChangeValidationResult
 984        {
 985            IsValid = issues.Count == 0,
 986            Issues = issues,
 987            Warnings = warnings
 988        };
 989    }
