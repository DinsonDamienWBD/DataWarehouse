 792    private static float CosineSimilarity(float[] a, float[] b)
 793    {
 794        if (a.Length != b.Length) return 0;
 795        float dot = 0, normA = 0, normB = 0;
 796        for (int i = 0; i < a.Length; i++)
 797        {
 798            dot += a[i] * b[i];
 799            normA += a[i] * a[i];
 800            normB += b[i] * b[i];
 801        }
 802        var denom = Math.Sqrt(normA) * Math.Sqrt(normB);
 803        return denom > 0 ? (float)(dot / denom) : 0;
 804    }
