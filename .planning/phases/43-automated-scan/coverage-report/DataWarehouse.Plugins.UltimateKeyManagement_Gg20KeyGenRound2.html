 487        /// <summary>
 488        /// GG20 Signing Round 3: Compute partial signature.
 489        /// </summary>
 490        public async Task<Gg20SignRound3> ProcessSignRound2Async(
 491            string keyId,
 492            Gg20SignRound2[] round2Messages,
 493            ISecurityContext context)
 494        {
 495            ValidateSecurityContext(context);
 496
 497            await _lock.WaitAsync();
 498            try
 499            {
 500                if (!_keys.TryGetValue(keyId, out var keyData))
 501                    throw new KeyNotFoundException($"Key '{keyId}' not found.");
 502
 503                var state = keyData.CurrentSigningState
 504                    ?? throw new InvalidOperationException("No active signing session.");
 505
 506                // Compute delta = sum of all delta_i
 507                var delta = state.GammaI.Multiply(state.Ki).Mod(DomainParams.N);
 508                foreach (var msg in round2Messages)
 509                {
 510                    if (msg.PartyIndex == _config.PartyIndex) continue;
 511                    delta = delta.Add(new BigInteger(1, msg.DeltaI)).Mod(DomainParams.N);
 512                }
 513
 514                // Compute k = delta^(-1) * sum(gamma_i) = 1/k (the full nonce inverse)
 515                var kInv = delta.ModInverse(DomainParams.N);
 516
 517                // Compute r = R.x mod n
 518                var r = state.R!.Normalize().AffineXCoord.ToBigInteger().Mod(DomainParams.N);
 519
 520                // Compute message hash as BigInteger
 521                var m = new BigInteger(1, state.MessageHash);
 522
 523                // Compute Lagrange coefficient
 524                var lambda = ComputeLagrangeCoefficient(_config.PartyIndex, state.Signers);
 525
 526                // Partial signature: s_i = k^(-1) * (m + r * lambda_i * x_i)
 527                var sI = kInv.Multiply(
 528                    m.Add(r.Multiply(lambda).Multiply(keyData.KeyShare).Mod(DomainParams.N))
 529                ).Mod(DomainParams.N);
 530
 531                state.Phase = 3;
 532                await PersistKeysToStorage();
 533
 534                return new Gg20SignRound3
 535                {
 536                    KeyId = keyId,
 537                    PartyIndex = _config.PartyIndex,
 538                    PartialS = sI.ToByteArrayUnsigned(),
 539                    R = r.ToByteArrayUnsigned()
 540                };
 541            }
 542            finally
 543            {
 544                _lock.Release();
 545            }
 546        }

 550        /// </summary>

 552            string keyId,

 554            ISecurityContext context)
 555        {

 597        private (BigInteger n, BigInteger lambda, BigInteger mu) GeneratePaillierKeys(int bits)
 598        {
 599            // Generate two safe primes p and q
 600            var p = GenerateSafePrime(bits / 2);
 601            var q = GenerateSafePrime(bits / 2);
 602
 603            var n = p.Multiply(q);
 604            var nSquared = n.Multiply(n);
 605
 606            // lambda = lcm(p-1, q-1)
 607            var pMinus1 = p.Subtract(BigInteger.One);
 608            var qMinus1 = q.Subtract(BigInteger.One);
 609            var lambda = pMinus1.Multiply(qMinus1).Divide(pMinus1.Gcd(qMinus1));
 610
 611            // g = n + 1 (simplified generator)
 612            var g = n.Add(BigInteger.One);
 613
 614            // mu = L(g^lambda mod n^2)^(-1) mod n
 615            // where L(u) = (u-1)/n
 616            var gLambda = g.ModPow(lambda, nSquared);
 617            var l = gLambda.Subtract(BigInteger.One).Divide(n);
 618            var mu = l.ModInverse(n);
 619
 620            return (n, lambda, mu);
 621        }

 623        private BigInteger PaillierEncrypt(BigInteger m, BigInteger n)
 624        {
 625            var nSquared = n.Multiply(n);
 626            var g = n.Add(BigInteger.One);
 627
 628            // Generate random r in Z*_n
 629            var r = GenerateRandomInRange(BigInteger.Two, n.Subtract(BigInteger.One));
 630
 631            // c = g^m * r^n mod n^2
 632            var gm = g.ModPow(m, nSquared);
 633            var rn = r.ModPow(n, nSquared);
 634            return gm.Multiply(rn).Mod(nSquared);
 635        }

 637        private BigInteger PaillierDecrypt(BigInteger c, BigInteger n, BigInteger lambda, BigInteger mu)
 638        {
 639            var nSquared = n.Multiply(n);
 640
 641            // L(c^lambda mod n^2) * mu mod n
 642            var cLambda = c.ModPow(lambda, nSquared);
 643            var l = cLambda.Subtract(BigInteger.One).Divide(n);
 644            return l.Multiply(mu).Mod(n);
 645        }

 648        {

 650            while (true)
 651            {
 652                var q = new BigInteger(bits - 1, 100, _secureRandom);
 653                var p = q.ShiftLeft(1).Add(BigInteger.One); // p = 2q + 1
 654
 655                if (p.IsProbablePrime(100))
 656                    return p;

 730        private bool VerifyFeldmanShare(BigInteger share, int partyIndex, ECPoint[] commitments)
 731        {
 732            var left = DomainParams.G.Multiply(share);

 734            var right = DomainParams.Curve.Infinity;

 747        private BigInteger ComputeLagrangeCoefficient(int i, int[] signers)
 748        {
 749            var num = BigInteger.One;
 750            var den = BigInteger.One;
 751
 752            foreach (var j in signers)
 753            {
 754                if (j == i) continue;
 755                num = num.Multiply(BigInteger.ValueOf(-j)).Mod(DomainParams.N);
 756                den = den.Multiply(BigInteger.ValueOf(i - j)).Mod(DomainParams.N);
 757            }
 758
 759            if (num.SignValue < 0)
 760                num = num.Add(DomainParams.N);
 761
 762            return num.Multiply(den.ModInverse(DomainParams.N)).Mod(DomainParams.N);
 763        }

 773        private static byte[] EncodeSignatureDer(BigInteger r, BigInteger s)
 774        {
 775            var rBytes = r.ToByteArrayUnsigned();
 776            var sBytes = s.ToByteArrayUnsigned();
 777
 778            if (rBytes[0] >= 0x80)
 779            {
 780                var tmp = new byte[rBytes.Length + 1];
 781                Array.Copy(rBytes, 0, tmp, 1, rBytes.Length);
 782                rBytes = tmp;
 783            }
 784            if (sBytes[0] >= 0x80)
 785            {
 786                var tmp = new byte[sBytes.Length + 1];
 787                Array.Copy(sBytes, 0, tmp, 1, sBytes.Length);
 788                sBytes = tmp;
 789            }
 790
 791            var totalLen = 4 + rBytes.Length + sBytes.Length;
 792            var result = new byte[2 + totalLen];
 793            var pos = 0;
 794
 795            result[pos++] = 0x30;
 796            result[pos++] = (byte)totalLen;
 797            result[pos++] = 0x02;
 798            result[pos++] = (byte)rBytes.Length;
 799            Array.Copy(rBytes, 0, result, pos, rBytes.Length);
 800            pos += rBytes.Length;
 801            result[pos++] = 0x02;
 802            result[pos++] = (byte)sBytes.Length;
 803            Array.Copy(sBytes, 0, result, pos, sBytes.Length);
 804
 805            return result;
 806        }
