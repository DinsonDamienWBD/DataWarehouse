 309                return false;
 310            }
 311
 312            // Step 3: Verify timestamp is valid
 313            if (payload.ValidUntil < DateTimeOffset.UtcNow)

 316                return false;

 318
 319            // Step 4: Verify Merkle root matches current state

 322            {

 324                if (proof != null)

 326                    currentProofs.Add(proof);
 327                }
 328            }

 330            var currentMerkleRoot = ComputeMerkleRoot(currentProofs);

 714    private string ComputeMerkleRoot(List<IntegrityProof> proofs)
 715    {
 716        if (proofs.Count == 0)
 717        {
 718            return string.Empty;
 719        }
 720
 721        // Compute leaf hashes
 722        var leaves = proofs
 723            .OrderBy(p => p.BlockId)
 724            .Select(p => ComputeProofHash(p))
 725            .ToList();
 726
 727        // Build Merkle tree
 728        while (leaves.Count > 1)
 729        {
 730            var newLevel = new List<byte[]>();
 731            for (int i = 0; i < leaves.Count; i += 2)
 732            {
 733                if (i + 1 < leaves.Count)
 734                {
 735                    var combined = new byte[leaves[i].Length + leaves[i + 1].Length];
 736                    Buffer.BlockCopy(leaves[i], 0, combined, 0, leaves[i].Length);
 737                    Buffer.BlockCopy(leaves[i + 1], 0, combined, leaves[i].Length, leaves[i + 1].Length);
 738                    // Hash computed inline; bus delegation to UltimateDataIntegrity available for centralized policy en
 739                    newLevel.Add(SHA256.HashData(combined));
 740                }
 741                else
 742                {
 743                    // Odd number of leaves - duplicate the last one
 744                    var combined = new byte[leaves[i].Length * 2];
 745                    Buffer.BlockCopy(leaves[i], 0, combined, 0, leaves[i].Length);
 746                    Buffer.BlockCopy(leaves[i], 0, combined, leaves[i].Length, leaves[i].Length);
 747                    // Hash computed inline; bus delegation to UltimateDataIntegrity available for centralized policy en
 748                    newLevel.Add(SHA256.HashData(combined));
 749                }
 750            }
 751            leaves = newLevel;
 752        }
 753
 754        return Convert.ToHexString(leaves[0]);
 755    }
