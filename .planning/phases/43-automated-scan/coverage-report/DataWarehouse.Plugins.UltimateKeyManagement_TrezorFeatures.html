 445        private byte[] EncodeMessage(ushort msgType, byte[] payload)
 446        {

 448            var message = new byte[6 + payload.Length];
 449            message[0] = (byte)'#';
 450            message[1] = (byte)'#';
 451            message[2] = (byte)(msgType >> 8);
 452            message[3] = (byte)(msgType & 0xFF);
 453            message[4] = (byte)(payload.Length >> 24);
 454            message[5] = (byte)(payload.Length >> 16);
 455            message[6 - 2] = (byte)(payload.Length >> 8);
 456            message[7 - 2] = (byte)(payload.Length & 0xFF);
 457
 458            // Fix: proper length encoding
 459            message[4] = (byte)((payload.Length >> 24) & 0xFF);
 460            message[5] = (byte)((payload.Length >> 16) & 0xFF);
 461            message[6] = (byte)((payload.Length >> 8) & 0xFF);
 462            message[7] = (byte)(payload.Length & 0xFF);
 463
 464            if (payload.Length > 0)

 466                // Resize array properly
 467                var fullMessage = new byte[8 + payload.Length];
 468                fullMessage[0] = (byte)'#';
