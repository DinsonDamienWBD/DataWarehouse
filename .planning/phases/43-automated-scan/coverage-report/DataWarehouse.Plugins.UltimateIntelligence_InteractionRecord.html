 416        queue.Enqueue((entity, 0));
 417
 418        while (queue.Count > 0)
 419        {
 420            var (current, level) = queue.Dequeue();
 421            if (level >= depth) continue;
 422
 423            if (_relationships.TryGetValue(current, out var related))
 424            {
 425                lock (related)
 426                {
 427                    foreach (var r in related.Where(r => visited.Add(r)))
 428                        queue.Enqueue((r, level + 1));
 429                }
 430            }
 431        }
 432
 433        visited.Remove(entity);
 434        return visited.ToList();
 435    }
 436}
