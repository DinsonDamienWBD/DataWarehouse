 958/// Handles autonomous decisions within policy bounds.
 959/// </summary>
 960public sealed class AutoDecision
 961{
 962    private readonly ConcurrentDictionary<string, AutoDecisionPolicy> _policies = new();
 963    private readonly ConcurrentQueue<DecisionRecord> _decisionHistory = new();
 964
 965    /// <summary>
 966    /// Registers a policy.
 967    /// </summary>
 968    /// <param name="policy">Policy to register.</param>
 969    public void RegisterPolicy(AutoDecisionPolicy policy)
 970    {
 971        _policies[policy.PolicyId] = policy;
 972    }
 973
 974    /// <summary>
 975    /// Evaluates if an action is allowed.
 976    /// </summary>
 977    /// <param name="action">Action to evaluate.</param>
 978    /// <param name="impactLevel">Estimated impact level.</param>
 979    /// <param name="policyId">Policy ID to use.</param>
 980    /// <returns>Decision result.</returns>
 981    public DecisionResult EvaluateAction(string action, float impactLevel, string? policyId = null)
 982    {
 983        var policy = policyId != null && _policies.TryGetValue(policyId, out var p)
 984            ? p
 985            : _policies.Values.FirstOrDefault();
 986
 987        if (policy == null)
 988        {
 989            return new DecisionResult
 990            {
 991                Action = action,
 992                IsAllowed = false,
 993                Reason = "No policy configured"
 994            };
 995        }
 996
 997        // Check denied list
 998        if (policy.DeniedActions.Any(d => action.Contains(d, StringComparison.OrdinalIgnoreCase)))
 999        {
 1000            return RecordDecision(new DecisionResult
 1001            {
 1002                Action = action,
 1003                IsAllowed = false,
 1004                Reason = "Action is in denied list",
 1005                PolicyId = policy.PolicyId
 1006            });
 1007        }
 1008
 1009        // Check impact level
 1010        if (impactLevel > policy.MaxImpactLevel)
 1011        {
 1012            return RecordDecision(new DecisionResult
 1013            {
 1014                Action = action,
 1015                IsAllowed = false,
 1016                Reason = $"Impact level {impactLevel} exceeds maximum {policy.MaxImpactLevel}",
 1017                PolicyId = policy.PolicyId
 1018            });
 1019        }
 1020
 1021        // Check allowed list (if not empty, action must be in it)
 1022        if (policy.AllowedActions.Count > 0 &&
 1023            !policy.AllowedActions.Any(a => action.Contains(a, StringComparison.OrdinalIgnoreCase)))
 1024        {
 1025            return RecordDecision(new DecisionResult
 1026            {
 1027                Action = action,
 1028                IsAllowed = false,
 1029                Reason = "Action not in allowed list",
 1030                PolicyId = policy.PolicyId
 1031            });
 1032        }
 1033
 1034        // Check if confirmation required
 1035        var needsConfirmation = policy.RequireConfirmation ||
 1036            policy.ConfirmationRequired.Any(c => action.Contains(c, StringComparison.OrdinalIgnoreCase));
 1037
 1038        return RecordDecision(new DecisionResult
 1039        {
 1040            Action = action,
 1041            IsAllowed = true,
 1042            RequiresConfirmation = needsConfirmation,
 1043            PolicyId = policy.PolicyId
 1044        });
 1045    }

 1047    private DecisionResult RecordDecision(DecisionResult result)
 1048    {
 1049        _decisionHistory.Enqueue(new DecisionRecord
 1050        {
 1051            Result = result,
 1052            Timestamp = DateTime.UtcNow
 1053        });
 1054
 1055        // Keep only last 1000 decisions
 1056        while (_decisionHistory.Count > 1000)
 1057            _decisionHistory.TryDequeue(out _);
 1058
 1059        return result;
 1060    }
