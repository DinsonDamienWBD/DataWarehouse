 507        private bool MatchesTarget(PbacPolicyTarget target, AccessContext context)
 508        {
 509            // Check subject match
 510            if (target.Subjects != null && target.Subjects.Any())
 511            {
 512                if (!target.Subjects.Contains("*") &&
 513                    !target.Subjects.Contains(context.SubjectId, StringComparer.OrdinalIgnoreCase) &&
 514                    !target.Subjects.Intersect(context.Roles, StringComparer.OrdinalIgnoreCase).Any())
 515                {
 516                    return false;
 517                }
 518            }
 519
 520            // Check resource match
 521            if (target.Resources != null && target.Resources.Any())
 522            {
 523                var matches = target.Resources.Any(r =>
 524                    r == "*" ||
 525                    context.ResourceId.Equals(r, StringComparison.OrdinalIgnoreCase) ||
 526                    (r.EndsWith("/*") && context.ResourceId.StartsWith(r[..^2], StringComparison.OrdinalIgnoreCase)) ||
 527                    Regex.IsMatch(context.ResourceId, r));
 528
 529                if (!matches) return false;
 530            }
 531
 532            // Check action match
 533            if (target.Actions != null && target.Actions.Any())
 534            {
 535                if (!target.Actions.Contains("*") &&
 536                    !target.Actions.Contains(context.Action, StringComparer.OrdinalIgnoreCase))
 537                {
 538                    return false;
 539                }
 540            }
 541
 542            return true;
 543        }

 545        private PolicyDecision CombineRuleDecisions(List<RuleEvaluationResult> results, RuleCombiningAlgorithm algorithm
 546        {
 547            return algorithm switch
 548            {
 549                RuleCombiningAlgorithm.DenyOverrides => results.Any(r => r.Effect == RuleEffect.Deny)
 550                    ? PolicyDecision.Deny
 551                    : results.Any(r => r.Effect == RuleEffect.Permit) ? PolicyDecision.Permit : PolicyDecision.NotApplic
 552
 553                RuleCombiningAlgorithm.PermitOverrides => results.Any(r => r.Effect == RuleEffect.Permit)
 554                    ? PolicyDecision.Permit
 555                    : results.Any(r => r.Effect == RuleEffect.Deny) ? PolicyDecision.Deny : PolicyDecision.NotApplicable
 556
 557                RuleCombiningAlgorithm.FirstApplicable => results.FirstOrDefault(r => r.Effect is RuleEffect.Permit or R
 558                {
 559                    RuleEffect.Permit => PolicyDecision.Permit,
 560                    RuleEffect.Deny => PolicyDecision.Deny,
 561                    _ => PolicyDecision.NotApplicable
 562                },
 563
 564                RuleCombiningAlgorithm.OnlyOneApplicable => results.Count(r => r.Effect is RuleEffect.Permit or RuleEffe
 565                    ? (results.Single(r => r.Effect is RuleEffect.Permit or RuleEffect.Deny).Effect == RuleEffect.Permit
 566                        ? PolicyDecision.Permit : PolicyDecision.Deny)
 567                    : PolicyDecision.Indeterminate,
 568
 569                _ => PolicyDecision.NotApplicable
 570            };
 571        }

 573        private AccessDecision ApplyConflictResolution(List<PolicyEvaluationDetail> evaluations, ConflictResolutionStrat
 574        {
 575            var applicableEvaluations = evaluations.Where(e => e.IsApplicable && e.Decision is PolicyDecision.Permit or 
 576
 577            if (!applicableEvaluations.Any())
 578            {
 579                return new AccessDecision
 580                {
 581                    IsGranted = false,
 582                    Reason = "No applicable policies found - access denied by default",
 583                    ApplicablePolicies = Array.Empty<string>()
 584                };
 585            }
 586
 587            bool isGranted;
 588            string reason;
 589            var applicablePolicies = applicableEvaluations.Select(e => e.PolicyId).ToArray();
 590
 591            switch (strategy)
 592            {
 593                case ConflictResolutionStrategy.DenyOverrides:
 594                    var denyPolicy = applicableEvaluations.FirstOrDefault(e => e.Decision == PolicyDecision.Deny);
 595                    if (denyPolicy != null)
 596                    {
 597                        isGranted = false;
 598                        reason = $"Denied by policy '{denyPolicy.PolicyName}' (deny-overrides)";
 599                    }
 600                    else
 601                    {
 602                        isGranted = true;
 603                        reason = "Permitted - no deny policies applicable";
 604                    }
 605                    break;
 606
 607                case ConflictResolutionStrategy.PermitOverrides:
 608                    var permitPolicy = applicableEvaluations.FirstOrDefault(e => e.Decision == PolicyDecision.Permit);
 609                    if (permitPolicy != null)
 610                    {
 611                        isGranted = true;
 612                        reason = $"Permitted by policy '{permitPolicy.PolicyName}' (permit-overrides)";
 613                    }
 614                    else
 615                    {
 616                        isGranted = false;
 617                        reason = "Denied - no permit policies applicable";
 618                    }
 619                    break;
 620
 621                case ConflictResolutionStrategy.FirstApplicable:
 622                    var firstApplicable = applicableEvaluations.OrderBy(e => e.Priority).First();
 623                    isGranted = firstApplicable.Decision == PolicyDecision.Permit;
 624                    reason = $"{(isGranted ? "Permitted" : "Denied")} by first applicable policy '{firstApplicable.Polic
 625                    break;
 626
 627                case ConflictResolutionStrategy.HighestPriority:
 628                    var highestPriority = applicableEvaluations.OrderBy(e => e.Priority).First();
 629                    isGranted = highestPriority.Decision == PolicyDecision.Permit;
 630                    reason = $"{(isGranted ? "Permitted" : "Denied")} by highest priority policy '{highestPriority.Polic
 631                    break;
 632
 633                default:
 634                    isGranted = false;
 635                    reason = "Unknown conflict resolution strategy";
 636                    break;
 637            }
 638
 639            return new AccessDecision
 640            {
 641                IsGranted = isGranted,
 642                Reason = reason,
 643                ApplicablePolicies = applicablePolicies,
 644                Metadata = new Dictionary<string, object>
 645                {
 646                    ["EvaluatedPolicies"] = evaluations.Count,
 647                    ["ApplicablePolicies"] = applicableEvaluations.Count,
 648                    ["ConflictResolution"] = strategy.ToString()
 649                }
 650            };
 651        }
