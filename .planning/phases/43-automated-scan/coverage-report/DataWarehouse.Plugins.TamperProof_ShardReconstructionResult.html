 481                var expectedHash = IntegrityHash.Create(manifest.HashAlgorithm, shardRecord.ContentHash);
 482                var actualHash = result.ActualHash != null
 483                    ? IntegrityHash.Create(manifest.HashAlgorithm, result.ActualHash)
 484                    : null;
 485
 486                if (result.LoadedSuccessfully && result.IntegrityValid)
 487                {
 488                    shards[result.ShardIndex] = result.Data;
 489                    shardResults.Add(ShardVerificationResult.CreateValid(result.ShardIndex, expectedHash, actualHash!));
 490                }
 491                else if (result.LoadedSuccessfully && !result.IntegrityValid)
 492                {
 493                    corruptedShards.Add(result.ShardIndex);
 494                    shardResults.Add(ShardVerificationResult.CreateFailed(
 495                        result.ShardIndex,
 496                        "Integrity verification failed",
 497                        expectedHash,
 498                        actualHash));
 499                }
 500                else
 501                {
 502                    missingShards.Add(result.ShardIndex);
 503                    shardResults.Add(ShardVerificationResult.CreateFailed(
 504                        result.ShardIndex,
 505                        result.ErrorMessage ?? "Shard unavailable"));
 506                }
 507            }
 508
 509            // Check if reconstruction is possible
 510            var needsReconstruction = corruptedShards.Count + missingShards.Count;
 511
 512            if (needsReconstruction > manifest.RaidConfiguration.ParityShardCount)
 513            {
 514                return new ShardReconstructionResult
 515                {
 516                    Success = false,
 517                    Data = null,
 518                    ShardResults = shardResults,
 519                    CorruptedShards = corruptedShards,
 520                    MissingShards = missingShards,
 521                    ReconstructedShards = new List<int>(),
 522                    ErrorMessage = $"Cannot reconstruct: {needsReconstruction} shards unavailable, " +
 523                                   $"only {manifest.RaidConfiguration.ParityShardCount} parity shards available"
 524                };
 525            }
 526
 527            // Reconstruct if needed
 528            var reconstructedShardIndices = new List<int>();
 529            if (needsReconstruction > 0)
 530            {
 531                ReconstructMissingShards(shards, manifest.RaidConfiguration);
 532                reconstructedShardIndices.AddRange(corruptedShards);
 533                reconstructedShardIndices.AddRange(missingShards);
 534            }
 535
 536            // Build final data
 537            var dataShards = shards.Take(manifest.RaidConfiguration.DataShardCount).ToList();
 538            var reconstructedData = new byte[manifest.FinalContentSize];
 539            var offset = 0;
 540
 541            for (int i = 0; i < dataShards.Count; i++)
 542            {
 543                var shard = dataShards[i];
 544                if (shard != null)
 545                {
 546                    var copyLength = Math.Min(shard.Length, reconstructedData.Length - offset);
 547                    Array.Copy(shard, 0, reconstructedData, offset, copyLength);
 548                    offset += copyLength;
 549                }
 550            }
 551
 552            return new ShardReconstructionResult
 553            {
 554                Success = true,
 555                Data = reconstructedData,
 556                ShardResults = shardResults,
 557                CorruptedShards = corruptedShards,
 558                MissingShards = missingShards,
 559                ReconstructedShards = reconstructedShardIndices
 560            };
 561        }
 562        catch (Exception ex)
 563        {
 564            logger.LogError(ex, "Failed to load and reconstruct shards with verification");
 565            return new ShardReconstructionResult
 566            {
 567                Success = false,
 568                Data = null,
 569                ShardResults = shardResults,
 570                CorruptedShards = corruptedShards,
 571                MissingShards = missingShards,
 572                ReconstructedShards = new List<int>(),
 573                ErrorMessage = ex.Message
 574            };
 575        }
 576    }

 578    /// <summary>
 579    /// Reconstructs missing shards using parity information (simple XOR for now).
 580    /// </summary>
 581    private static void ReconstructMissingShards(List<byte[]?> shards, RaidRecord raidConfig)
 582    {
 583        // Simple XOR reconstruction for demonstration
 584        // Production would use Reed-Solomon codes
 585
 586        for (int i = 0; i < raidConfig.DataShardCount; i++)
 587        {
 588            if (shards[i] == null)
 589            {
 590                // Reconstruct data shard from other data shards and parity
 591                var shardSize = raidConfig.ShardSize;
 592                var reconstructed = new byte[shardSize];
 593
 594                // XOR all available data shards and first parity shard
 595                for (int j = 0; j < raidConfig.DataShardCount; j++)
 596                {
 597                    if (j != i && shards[j] != null)
 598                    {
 599                        for (int k = 0; k < Math.Min(shardSize, shards[j]!.Length); k++)
 600                        {
 601                            reconstructed[k] ^= shards[j]![k];
 602                        }
 603                    }
 604                }
 605
 606                // XOR with first parity shard
 607                var parityShard = shards[raidConfig.DataShardCount];
 608                if (parityShard != null)
 609                {
 610                    for (int k = 0; k < Math.Min(shardSize, parityShard.Length); k++)
 611                    {
 612                        reconstructed[k] ^= parityShard[k];
 613                    }
 614                }
 615
 616                shards[i] = reconstructed;
 617            }
 618        }
 619    }

 621    /// <summary>
 622    /// Phase 3: Verify integrity of reconstructed data.
 623    /// </summary>
 624    /// <param name="data">Reconstructed data to verify.</param>
 625    /// <param name="manifest">Manifest with expected hashes.</param>
 626    /// <param name="readMode">Verification level.</param>
 627    /// <param name="integrity">Integrity provider.</param>

 671
 672                return IntegrityVerificationResult.CreateFailed(

 676            }
 677        }
 678        catch (Exception ex)
 679        {

 682        }

 685    /// <summary>
 686    /// Phase 3b: Recover data from WORM backup when tampering is detected.
 687    /// </summary>
 688    /// <param name="manifest">Manifest with WORM reference.</param>
 689    /// <param name="worm">WORM storage provider.</param>
 690    /// <param name="dataStorage">Data tier storage for restoring shards.</param>
 691    /// <param name="logger">Logger instance.</param>
 692    /// <param name="ct">Cancellation token.</param>
 693    /// <returns>Recovery result.</returns>
 694    public static async Task<RecoveryResult> RecoverFromWormAsync(
 695        TamperProofManifest manifest,
 696        IWormStorageProvider worm,
 697        IStorageProvider dataStorage,
 698        ILogger logger,
 699        CancellationToken ct)
 700    {
 701        logger.LogWarning("Attempting recovery from WORM for object {ObjectId}", manifest.ObjectId);
 702
 703        try
 704        {
 705            if (manifest.WormBackup == null)
 706            {
 707                return RecoveryResult.CreateFailure(
 708                    manifest.ObjectId,
 709                    manifest.Version,
 710                    "No WORM backup reference in manifest");
 711            }
 712
 713            // Load data from WORM
 714            var wormData = await worm.ReadAsync(manifest.WormBackup.StorageLocation, ct);
 715
 716            if (wormData == null)
 717            {
 718                return RecoveryResult.CreateFailure(
 719                    manifest.ObjectId,
 720                    manifest.Version,
 721                    "WORM backup not found");
 722            }
 723
 724            // Restore shards to data storage
 725            logger.LogDebug("Restoring shards from WORM backup to data storage");
 726
 727            try
 728            {
 729                // Recreate RAID shards from WORM backup
 730                var shardSize = manifest.RaidConfiguration.ShardSize;
 731                var dataShardCount = manifest.RaidConfiguration.DataShardCount;
 732                var parityShardCount = manifest.RaidConfiguration.ParityShardCount;
 733
 734                // Split WORM data back into data shards
 735                var shards = new List<byte[]>();
 736                for (int i = 0; i < dataShardCount; i++)
 737                {
 738                    var start = i * shardSize;
 739                    var length = Math.Min(shardSize, wormData.Length - start);
 740                    if (length > 0)
 741                    {
 742                        var shard = new byte[length];
 743                        Array.Copy(wormData, start, shard, 0, length);
 744                        shards.Add(shard);
 745                    }
 746                }
 747
 748                // Recreate parity shards (simple XOR for now)
 749                for (int i = 0; i < parityShardCount; i++)
 750                {
 751                    var parityShard = new byte[shardSize];
 752                    for (int j = 0; j < dataShardCount; j++)
 753                    {
 754                        for (int k = 0; k < Math.Min(shardSize, shards[j].Length); k++)
 755                        {
 756                            parityShard[k] ^= shards[j][k];
 757                        }
 758                    }
 759                    shards.Add(parityShard);
 760                }
 761
 762                // Write shards back to data storage
 763                var writeErrors = new List<string>();
 764                var writtenCount = 0;
 765
 766                for (int i = 0; i < shards.Count; i++)
 767                {
 768                    var shardRecord = manifest.RaidConfiguration.Shards[i];
 769                    var uri = new Uri($"data://shards/{shardRecord.StorageLocation}");
 770
 771                    try
 772                    {
 773                        using var shardStream = new MemoryStream(shards[i]);
 774                        await dataStorage.SaveAsync(uri, shardStream);
 775                        writtenCount++;
 776                        logger.LogDebug("Restored shard {Index} to {Location}", i, shardRecord.StorageLocation);
 777                    }
 778                    catch (Exception ex)
 779                    {
 780                        writeErrors.Add($"Shard {i}: {ex.Message}");
 781                        logger.LogWarning(ex, "Failed to restore shard {Index}", i);
 782                    }
 783                }
 784
 785                if (writeErrors.Count > 0)
 786                {
 787                    return RecoveryResult.CreateFailure(
 788                        manifest.ObjectId,
 789                        manifest.Version,
 790                        $"Partial restoration: {writtenCount}/{shards.Count} shards restored. Errors: {string.Join("; ",
 791                }
 792
 793                logger.LogInformation("Recovery from WORM successful: restored {Count} shards for object {ObjectId}",
 794                    writtenCount, manifest.ObjectId);
 795
 796                return RecoveryResult.CreateSuccess(
 797                    manifest.ObjectId,
 798                    manifest.Version,
 799                    "WORM Storage",
 800                    $"Restored {writtenCount} shards from immutable backup");
 801            }
 802            catch (Exception ex)
 803            {
 804                logger.LogError(ex, "Failed to restore shards from WORM backup");
 805                return RecoveryResult.CreateFailure(
 806                    manifest.ObjectId,
 807                    manifest.Version,
 808                    $"Shard restoration error: {ex.Message}");
 809            }
 810        }
 811        catch (Exception ex)
 812        {
 813            logger.LogError(ex, "WORM recovery failed for object {ObjectId}", manifest.ObjectId);
 814            return RecoveryResult.CreateFailure(
 815                manifest.ObjectId,
 816                manifest.Version,
 817                $"Recovery error: {ex.Message}");
 818        }
 819    }

 821    /// <summary>
 822    /// Phase 4: Reverse pipeline transformations to get original data.
 823    /// </summary>
 824    /// <param name="transformedData">Data after pipeline transformations.</param>
 825    /// <param name="manifest">Manifest with pipeline stage history.</param>
 826    /// <param name="orchestrator">Pipeline orchestrator.</param>
 827    /// <param name="logger">Logger instance.</param>
 828    /// <param name="ct">Cancellation token.</param>
 829    /// <returns>Original data before transformations.</returns>
 830    public static async Task<byte[]> ReversePipelineTransformationsAsync(
 831        byte[] transformedData,
 832        TamperProofManifest manifest,
 833        IPipelineOrchestrator orchestrator,
 834        ILogger logger,
 835        CancellationToken ct)
 836    {
 837        logger.LogDebug("Reversing {Count} pipeline transformations", manifest.PipelineStages.Count);
 838
 839        try
 840        {
 841            var currentData = transformedData;
 842
 843            // Apply reverse transformations if pipeline stages were used
 844            if (manifest.PipelineStages.Count > 0)
 845            {
 846                logger.LogDebug("Applying reverse transformations for {Count} pipeline stages", manifest.PipelineStages.
 847
 848                // Use orchestrator to reverse the pipeline
 849                // The orchestrator should handle reversing in the correct order
 850                using var inputStream = new MemoryStream(currentData);
 851                using var outputStream = await orchestrator.ReversePipelineAsync(inputStream, ct);
 852
 853                using var resultMs = new MemoryStream();
 854                await outputStream.CopyToAsync(resultMs, ct);
 855                currentData = resultMs.ToArray();
 856
 857                logger.LogDebug("Pipeline reversal complete: transformed {FromBytes} bytes to {ToBytes} bytes",
 858                    transformedData.Length, currentData.Length);
 859
 860                // Verify intermediate hashes if needed (for debugging/auditing)
 861                for (int i = manifest.PipelineStages.Count - 1; i >= 0; i--)
 862                {
 863                    var stage = manifest.PipelineStages[i];
 864                    logger.LogDebug("Reversed stage {Index}: {Type} ({InputSize} -> {OutputSize} bytes)",
 865                        stage.StageIndex, stage.StageType, stage.OutputSize, stage.InputSize);
 866                }
 867            }
 868            else
 869            {
 870                logger.LogDebug("No pipeline stages to reverse");
 871            }
 872
 873            // Remove content padding if present
 874            if (manifest.ContentPadding != null)
 875            {
 876                logger.LogDebug("Removing content padding: {Prefix} prefix + {Suffix} suffix bytes",
 877                    manifest.ContentPadding.PrefixPaddingBytes,
 878                    manifest.ContentPadding.SuffixPaddingBytes);
 879
 880                var unpadded = new byte[currentData.Length - manifest.ContentPadding.TotalPaddingBytes];
 881                Array.Copy(
 882                    currentData,
 883                    manifest.ContentPadding.PrefixPaddingBytes,
 884                    unpadded,
 885                    0,
 886                    unpadded.Length);
 887
 888                currentData = unpadded;
 889            }
 890
 891            logger.LogDebug("Pipeline reversal complete: {Bytes} bytes", currentData.Length);
 892
 893            return currentData;
 894        }
 895        catch (Exception ex)
 896        {
 897            logger.LogError(ex, "Failed to reverse pipeline transformations");
 898            throw new InvalidOperationException("Pipeline reversal failed", ex);
 899        }
 900    }
