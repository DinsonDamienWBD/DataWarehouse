 608        {

 610            var buffer = new byte[bufferSize];

 619                var bytesRead = await data.ReadAsync(buffer.AsMemory(0, toRead), ct);
 620
 621                if (bytesRead == 0)
 622                {
 623                    break;

 626                // Write to memory-mapped file
 627                accessor.WriteArray(position, buffer, 0, bytesRead);
 628                position += bytesRead;
 629                totalWritten += bytesRead;
 630
 631                // Periodic flush for large files
 632                if (totalWritten % (16 * 1024 * 1024) == 0) // Every 16MB
 633                {
 634                    await FlushWithStrategyAsync(accessor, position - bytesRead, bytesRead);
 635                }
 636            }
 637
 638            return totalWritten;
 639        }
