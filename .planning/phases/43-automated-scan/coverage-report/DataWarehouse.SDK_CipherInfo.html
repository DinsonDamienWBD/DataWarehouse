 889        /// <summary>
 890        /// Serializes the payload to a byte array.
 891        /// Format: [Version:1][AlgIdLen:1][AlgId:var][Nonce:var][TagLen:2][Tag:var][CiphertextLen:4][Ciphertext:var]
 892        /// </summary>
 893        public byte[] ToBytes()
 894        {
 895            using var ms = new System.IO.MemoryStream();
 896            using var writer = new System.IO.BinaryWriter(ms, System.Text.Encoding.UTF8, leaveOpen: true);
 897
 898            writer.Write((byte)Version);
 899            var algBytes = System.Text.Encoding.UTF8.GetBytes(AlgorithmId);
 900            writer.Write((byte)algBytes.Length);
 901            writer.Write(algBytes);

 903            writer.Write(Nonce);
 904            writer.Write((ushort)Tag.Length);
 905            writer.Write(Tag);

 909            // Optional fields

 911            var hasWrappedDek = WrappedDek != null && WrappedDek.Length > 0;
 912            byte flags = 0;
 913            if (hasKeyId) flags |= 0x01;
 914            if (hasWrappedDek) flags |= 0x02;

 916
 917            if (hasKeyId)
 918            {
 919                var keyIdBytes = System.Text.Encoding.UTF8.GetBytes(KeyId!);
 920                writer.Write((byte)keyIdBytes.Length);
 921                writer.Write(keyIdBytes);
 922            }
 923
 924            if (hasWrappedDek)
 925            {
 926                writer.Write((ushort)WrappedDek!.Length);
 927                writer.Write(WrappedDek);
 928                if (!string.IsNullOrEmpty(KekId))
 929                {
 930                    var kekIdBytes = System.Text.Encoding.UTF8.GetBytes(KekId);
 931                    writer.Write((byte)kekIdBytes.Length);
 932                    writer.Write(kekIdBytes);
 933                }
 934                else
 935                {
 936                    writer.Write((byte)0);
 937                }
 938            }
 939
 940            return ms.ToArray();
 941        }
