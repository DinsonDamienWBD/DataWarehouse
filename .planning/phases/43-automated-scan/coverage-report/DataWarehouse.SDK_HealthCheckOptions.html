 793    private void CheckMemoryPressure(object? state)
 794    {

 812                {
 813                    RequestCollection(aggressive: newLevel == MemoryPressureLevel.Critical);
 814                }
 815            }
 816        }
 817        catch
 818        {
 819            // Ignore monitoring errors
 820        }
 821    }

 827        if (usagePercent >= _elevatedThreshold) return MemoryPressureLevel.Elevated;

 840        {
 841            try
 842            {
 843                await handler(level).ConfigureAwait(false);
 844            }
 845            catch
 846            {
 847                // Ignore handler errors

 850    }
