 881/// <summary>
 882/// 91.G1: Parallel parity calculator using multi-threaded computation.
 883/// Distributes parity calculation across available CPU cores for large stripe units.
 884/// </summary>
 885public sealed class ParallelParityCalculator
 886{
 887    private readonly int _maxDegreeOfParallelism;
 888    private readonly int _minChunkSizeForParallel;
 889    private long _totalCalculations;
 890    private long _parallelCalculations;

 893    /// <summary>
 894    /// Creates a new parallel parity calculator.
 895    /// </summary>
 896    /// <param name="maxDegreeOfParallelism">Maximum threads for parallel calculation. Defaults to processor count.</par
 897    /// <param name="minChunkSizeForParallel">Minimum data size in bytes before parallel execution is used. Default 64KB
 898    public ParallelParityCalculator(int? maxDegreeOfParallelism = null, int minChunkSizeForParallel = 65536)
 899    {

 901        _minChunkSizeForParallel = minChunkSizeForParallel;

 904    /// <summary>
 905    /// Calculates XOR parity across multiple data chunks using parallel threads.
 906    /// Each thread processes a segment of the byte range independently.
 907    /// </summary>
 908    public byte[] CalculateXorParity(IReadOnlyList<byte[]> dataChunks)
 909    {
 910        if (dataChunks.Count == 0) return Array.Empty<byte>();
 911
 912        Interlocked.Increment(ref _totalCalculations);
 913        var length = dataChunks[0].Length;
 914        var result = new byte[length];
 915
 916        if (length < _minChunkSizeForParallel || dataChunks.Count < 2)
 917        {
 918            // Sequential fallback for small data
 919            CalculateXorSequential(dataChunks, result, 0, length);
 920        }
 921        else
 922        {
 923            Interlocked.Increment(ref _parallelCalculations);
 924            var segmentSize = (length + _maxDegreeOfParallelism - 1) / _maxDegreeOfParallelism;
 925
 926            Parallel.For(0, _maxDegreeOfParallelism, new ParallelOptions { MaxDegreeOfParallelism = _maxDegreeOfParallel
 927            {
 928                var start = segment * segmentSize;
 929                var end = Math.Min(start + segmentSize, length);
 930                if (start < length)
 931                {
 932                    CalculateXorSequential(dataChunks, result, start, end);
 933                }
 934            });
 935        }
 936
 937        Interlocked.Add(ref _totalBytesProcessed, length * dataChunks.Count);
 938        return result;
 939    }

 942    /// Calculates XOR parity for a range using SIMD acceleration when available.

 945    {

 947        Array.Copy(chunks[0], start, result, start, end - start);
 948

 950        for (int c = 1; c < chunks.Count; c++)

 973
 974            Parallel.For(0, _maxDegreeOfParallelism, new ParallelOptions { MaxDegreeOfParallelism = _maxDegreeOfParallel
 975            {

 978                if (start < length)

 988    private void CalculateQParityRange(IReadOnlyList<byte[]> chunks, byte[] result, int start, int end)
 989    {
 990        for (int i = start; i < end; i++)
 991        {
 992            byte value = 0;
 993            for (int j = 0; j < chunks.Count; j++)
 994            {
 995                byte coefficient = (byte)(1 << (j % 8));
 996                value ^= GaloisMultiply(chunks[j][i], coefficient);
 997            }
 998            result[i] = value;
 999        }
 1000    }
 1001
 1002    private static byte GaloisMultiply(byte a, byte b)
 1003    {
 1004        byte result = 0;
 1005        byte temp = a;
 1006
 1007        for (int i = 0; i < 8; i++)
 1008        {
 1009            if ((b & 1) != 0)
 1010                result ^= temp;
 1011
 1012            bool highBitSet = (temp & 0x80) != 0;
 1013            temp <<= 1;
 1014
 1015            if (highBitSet)
 1016                temp ^= 0x1D; // Primitive polynomial: x^8 + x^4 + x^3 + x^2 + 1
 1017
