    private static IBlockDevice CreateIoUring(string path, int blockSize, long blockCount, bool createNew,
        BlockDeviceOptions options, ILogger? logger)
    {
        if (!IO.Linux.IoUringBindings.IsAvailable)
            throw new PlatformNotSupportedException("io_uring is not available. Requires Linux 5.1+ with liburing.");
        logger?.LogInformation("BlockDeviceFactory: forced IoUring");
        return new IO.Linux.IoUringBlockDevice(path, blockSize, blockCount, options);
    }

    private static IBlockDevice CreateIoRing(string path, int blockSize, long blockCount, bool createNew,
        BlockDeviceOptions options, ILogger? logger)
    {
        var device = IO.Windows.IoRingBlockDevice.TryCreate(path, blockSize, blockCount, createNew, options);
        if (device == null)
            throw new PlatformNotSupportedException("IoRing is not available. Requires Windows 11+.");
        logger?.LogInformation("BlockDeviceFactory: forced IoRing");
        return device;
    }

    private static IBlockDevice CreateWindowsOverlapped(string path, int blockSize, long blockCount, bool createNew,
        BlockDeviceOptions options, ILogger? logger)
    {
        var device = IO.Windows.WindowsOverlappedBlockDevice.TryCreate(path, blockSize, blockCount, createNew, options);
        if (device == null)
            throw new PlatformNotSupportedException("Windows Overlapped I/O is not available. Requires Windows.");

        if (device == null)

        logger?.LogInformation("BlockDeviceFactory: forced Kqueue");
