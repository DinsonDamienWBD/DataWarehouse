    ```csharp
    /// <summary>
    /// Attempts to create a kqueue-backed block device. Returns null if kqueue is not
    /// available (non-macOS/FreeBSD platform or kqueue syscall unavailable).
    /// </summary>
    /// <param name="path">Path to the container file.</param>
    /// <param name="blockSize">Block size in bytes.</param>
    /// <param name="blockCount">Total number of blocks.</param>
    /// <param name="createNew">If true, creates and pre-allocates file.</param>
    /// <param name="options">Optional block device options.</param>
    /// <returns>A KqueueBlockDevice instance, or null if not supported.</returns>
    public static KqueueBlockDevice? TryCreate(
        string path, int blockSize, long blockCount, bool createNew,
        BlockDeviceOptions? options = null)
    {
        if (!IsSupported) return null;

        try
        {
            return new KqueueBlockDevice(path, blockSize, blockCount, createNew);
        }
        catch (PlatformNotSupportedException)
        {
            return null;
        }
        catch (IOException)
        {
            // kqueue fd creation or file open failed -- not usable
            return null;
        }
    }
    ```

    Place it right after the existing `IsSupported` property (around line 187).

    Do NOT modify any existing read/write/batch/flush/dispose logic. Only add the TryCreate method.
