     ```csharp
     private static IEnumerable<byte[]> GeneratePayload(long totalBytes, int chunkSize = 1_048_576) // 1MB chunks
     {
         long remaining = totalBytes;
         var rng = new Random(42); // deterministic seed for reproducibility
         while (remaining > 0)
         {
             int size = (int)Math.Min(chunkSize, remaining);
             var chunk = new byte[size];
             rng.NextBytes(chunk);
             remaining -= size;
             yield return chunk;
         }
     }
     ```

2. Test methods:
